104  if ((jsgf_path = getenv("JSGF_PATH")) != NULL) {
105  char *word, *c;
106 
107  /* FIXME: This should be a function in libsphinxbase. */
108  /* FIXME: Also nextword() is totally useless... */
109  word = jsgf_path = ckd_salloc(jsgf_path);
110  while ((c = strchr(word, ':'))) {
111  *c = '\0';
112  grammar->searchpath = glist_add_ptr(grammar->searchpath, word);
113  word = c + 1;
114  }
115  grammar->searchpath = glist_add_ptr(grammar->searchpath, word);
116  grammar->searchpath = glist_reverse(grammar->searchpath);

315 static int
316 expand_rhs(jsgf_t *grammar, jsgf_rule_t *rule, jsgf_rhs_t *rhs,
317  int rule_entry, int rule_exit)
318 {
319  gnode_t *gn;
320  int lastnode;
321 
322  /* Last node expanded in this sequence. */
323  lastnode = rule_entry;
324 
325  /* Iterate over atoms in rhs and generate links/nodes */
326  for (gn = rhs->atoms; gn; gn = gnode_next(gn)) {
327  jsgf_atom_t *atom = gnode_ptr(gn);
328 
329  if (jsgf_atom_is_rule(atom)) {
330  jsgf_rule_t *subrule;
331  char *fullname;
332  gnode_t *subnode;
333  jsgf_rule_stack_t *rule_stack_entry;
334 
335  /* Special case for <NULL> and <VOID> pseudo-rules
336  If this is the only atom in the rhs, and it's the
337  first rhs in the rule, then emit a null transition,
338  creating an exit state if needed. */
339  if (0 == strcmp(atom->name, "<NULL>")) {
340  if (gn == rhs->atoms && gnode_next(gn) == NULL) {
341  if (rule_exit == NO_NODE) {
342  jsgf_add_link(grammar, atom,
343  lastnode, grammar->nstate);
344  rule_exit = lastnode = grammar->nstate;
345  ++grammar->nstate;
346  } else {
347  jsgf_add_link(grammar, atom,
348  lastnode, rule_exit);
349  }
350  }
351  continue;
352  }
353  else if (0 == strcmp(atom->name, "<VOID>")) {
354  /* Make this entire RHS unspeakable */
355  return NO_NODE;
356  }
357 
358  fullname = jsgf_fullname_from_rule(rule, atom->name);
359  if (hash_table_lookup(grammar->rules, fullname, (void**)&subrule) == -1) {
360  E_ERROR("Undefined rule in RHS: %s\n", fullname);
361  ckd_free(fullname);
362  return NO_NODE;
363  }
364  ckd_free(fullname);
365 
366  /* Look for this subrule in the stack of expanded rules */
367  for (subnode = grammar->rulestack; subnode; subnode = gnode_next(subnode)) {
368  rule_stack_entry = (jsgf_rule_stack_t *)gnode_ptr(subnode);
369  if (rule_stack_entry->rule == subrule)
370  break;
371  }
372 
373  if (subnode != NULL) {
374  /* Allow right-recursion only. */
375  if (gnode_next(gn) != NULL) {
376  E_ERROR("Only right-recursion is permitted (in %s.%s)\n",
377  grammar->name, rule->name);
378  return NO_NODE;
379  }
380  /* Add a link back to the beginning of this rule instance */
381  E_INFO("Right recursion %s %d => %d\n", atom->name, lastnode, rule_stack_entry->entry);
382  jsgf_add_link(grammar, atom, lastnode, rule_stack_entry->entry);
383 
384  /* Let our caller know that this rhs didn't reach an
385  end state. */
386  lastnode = RECURSIVE_NODE;
387  }
388  else {
389  /* If this is the last atom in this rhs, link its
390  expansion to the parent rule's exit state.
391  Otherwise, create a new exit state for it. */
392  int subruleexit = NO_NODE;
393  if (gnode_next(gn) == NULL && rule_exit >= 0)
394  subruleexit = rule_exit;
395 
396  /* Expand the subrule */
397  lastnode = expand_rule(grammar, subrule, lastnode, subruleexit);
398 
399  if (lastnode == NO_NODE)
400  return NO_NODE;
401  }
402  }
403  else {
404  /* An exit-state is created if this isn't the last atom
405  in the rhs, or if the containing rule doesn't have an
406  exit state yet.
407  Otherwise, the rhs's exit state becomes the containing
408  rule's exit state. */
409  int exitstate;
410  if (gnode_next(gn) == NULL && rule_exit >= 0) {
411  exitstate = rule_exit;
412  } else {
413  exitstate = grammar->nstate;
414  ++grammar->nstate;
415  }
416 
417  /* Add a link for this token */
418  jsgf_add_link(grammar, atom,
419  lastnode, exitstate);
420  lastnode = exitstate;
421  }
422  }
423 
424  return lastnode;
425 }

427 static int
428 expand_rule(jsgf_t *grammar, jsgf_rule_t *rule, int rule_entry,
429  int rule_exit)
430 {
431  jsgf_rule_stack_t* rule_stack_entry;
432  jsgf_rhs_t *rhs;
433  float norm;
434 
435  /* Push this rule onto the stack */
436  rule_stack_entry = (jsgf_rule_stack_t*)ckd_calloc(1, sizeof (jsgf_rule_stack_t));
437  rule_stack_entry->rule = rule;
438  rule_stack_entry->entry = rule_entry;
439  grammar->rulestack = glist_add_ptr(grammar->rulestack,
440  rule_stack_entry);
441 
442  /* Normalize weights for all alternatives exiting rule_entry */
443  norm = 0;
444  for (rhs = rule->rhs; rhs; rhs = rhs->alt) {
445  if (rhs->atoms) {
446  jsgf_atom_t *atom = gnode_ptr(rhs->atoms);
447  norm += atom->weight;
448  }
449  }
450 
451  if (norm == 0) norm = 1;
452  for (rhs = rule->rhs; rhs; rhs = rhs->alt) {
453  int lastnode;
454 
455  if (rhs->atoms) {
456  jsgf_atom_t *atom = gnode_ptr(rhs->atoms);
457  atom->weight /= norm;
458  }
459 
460  lastnode = expand_rhs(grammar, rule, rhs,
461  rule_entry, rule_exit);
462 
463  if (lastnode == NO_NODE) {
464  return NO_NODE;
465  } else if (lastnode == RECURSIVE_NODE) {
466  /* The rhs ended with right-recursion, i.e. a transition to
467  an earlier state. Nothing needs to happen at this level. */
468  ;
469  } else if (rule_exit == NO_NODE) {
470  /* If this rule doesn't have an exit state yet, use the exit
471  state of its first right-hand-side.
472  All other right-hand-sides will use this exit state. */
473  assert (lastnode >= 0);
474  rule_exit = lastnode;
475  }
476  }
477 
478  /* If no exit-state was created, use the entry-state. */
479  if (rule_exit == NO_NODE) {
480  rule_exit = rule_entry;
481  }
482 
483  /* Pop this rule from the rule stack */
484  ckd_free(gnode_ptr(grammar->rulestack));
485  grammar->rulestack = gnode_free(grammar->rulestack, NULL);
486 
487  return rule_exit;
488 }

518 static fsg_model_t *
519 jsgf_build_fsg_internal(jsgf_t *grammar, jsgf_rule_t *rule,
520  logmath_t *lmath, float32 lw, int do_closure)
521 {
522  fsg_model_t *fsg;
523  glist_t nulls;
524  gnode_t *gn;
525  int rule_entry, rule_exit;
526 
527  /* Clear previous links */
528  for (gn = grammar->links; gn; gn = gnode_next(gn)) {
529  ckd_free(gnode_ptr(gn));
530  }
531  glist_free(grammar->links);
532  grammar->links = NULL;
533  grammar->nstate = 0;
534 
535  /* Create the top-level entry state, and expand the
536  top-level rule. */
537  rule_entry = grammar->nstate++;
538  rule_exit = expand_rule(grammar, rule, rule_entry, NO_NODE);
539 
540  /* If no exit-state was created, create one. */
541  if (rule_exit == NO_NODE) {
542  rule_exit = grammar->nstate++;
543  jsgf_add_link(grammar, NULL, rule_entry, rule_exit);
544  }
545 
546  fsg = fsg_model_init(rule->name, lmath, lw, grammar->nstate);
547  fsg->start_state = rule_entry;
548  fsg->final_state = rule_exit;
549  grammar->links = glist_reverse(grammar->links);
550  for (gn = grammar->links; gn; gn = gnode_next(gn)) {
551  jsgf_link_t *link = gnode_ptr(gn);
552 
553  if (link->atom) {
554  if (jsgf_atom_is_rule(link->atom)) {
555  fsg_model_null_trans_add(fsg, link->from, link->to,
556  logmath_log(lmath, link->atom->weight));
557  }
558  else {
559  int wid = fsg_model_word_add(fsg, link->atom->name);
560  fsg_model_trans_add(fsg, link->from, link->to,
561  logmath_log(lmath, link->atom->weight), wid);
562  }
563  }
564  else {
565  fsg_model_null_trans_add(fsg, link->from, link->to, 0);
566  }
567  }
568  if (do_closure) {
569  nulls = fsg_model_null_trans_closure(fsg, NULL);
570  glist_free(nulls);
571  }
572 
573  return fsg;
574 }

591 jsgf_read_file(const char *file, logmath_t * lmath, float32 lw)
592 {

600  return NULL;
601  }
602 
603  rule = NULL;
604  for (itor = jsgf_rule_iter(jsgf); itor;
605  itor = jsgf_rule_iter_next(itor)) {

614  return NULL;
615  }
616  fsg = jsgf_build_fsg(jsgf, rule, lmath, lw);
617  jsgf_grammar_free(jsgf);
618  return fsg;

640 jsgf_define_rule(jsgf_t *jsgf, char *name, jsgf_rhs_t *rhs, int is_public)
641 {
642  jsgf_rule_t *rule;
643  void *val;
644 
645  if (name == NULL) {
646  name = ckd_malloc(strlen(jsgf->name) + 16);
647  sprintf(name, "<%s.g%05d>", jsgf->name, hash_table_inuse(jsgf->rules));
648  }
649  else {
650  char *newname;
651 
652  newname = jsgf_fullname(jsgf, name);
653  name = newname;
654  }
655 
656  rule = ckd_calloc(1, sizeof(*rule));
657  rule->refcnt = 1;
658  rule->name = ckd_salloc(name);
659  rule->rhs = rhs;

715 jsgf_rule_t *
716 jsgf_import_rule(jsgf_t *jsgf, char *name)
717 {
718  char *c, *path, *newpath;
719  size_t namelen, packlen;
720  void *val;
721  jsgf_t *imp;
722  int import_all;
723 
724  /* Trim the leading and trailing <> */
725  namelen = strlen(name);
726  path = ckd_malloc(namelen - 2 + 6); /* room for a trailing .gram */
727  strcpy(path, name + 1);
728  /* Split off the first part of the name */
729  c = strrchr(path, '.');
730  if (c == NULL) {
731  E_ERROR("Imported rule is not qualified: %s\n", name);
732  ckd_free(path);
733  return NULL;
734  }
735  packlen = c - path;
736  *c = '\0';
737 
738  /* Look for import foo.* */
739  import_all = (strlen(name) > 2 && 0 == strcmp(name + namelen - 3, ".*>"));
740 
741  /* Construct a filename. */
742  for (c = path; *c; ++c)
743  if (*c == '.') *c = '/';
744  strcat(path, ".gram");
745  newpath = path_list_search(jsgf->searchpath, path);
746  ckd_free(path);
747  if (newpath == NULL)
748  return NULL;
749 
750  path = newpath;
751  E_INFO("Importing %s from %s to %s\n", name, path, jsgf->name);
752 
753  /* FIXME: Also, we need to make sure that path is fully qualified
754  * here, by adding any prefixes from jsgf->name to it. */
755  /* See if we have parsed it already */
756  if (hash_table_lookup(jsgf->imports, path, &val) == 0) {
757  E_INFO("Already imported %s\n", path);
758  imp = val;
759  ckd_free(path);
760  }
761  else {
762  /* If not, parse it. */
763  imp = jsgf_parse_file(path, jsgf);
764  val = hash_table_enter(jsgf->imports, path, imp);
765  if (val != (void *)imp) {
766  E_WARN("Multiply imported file: %s\n", path);
767  }
768  }
769  if (imp != NULL) {
770  hash_iter_t *itor;
771  /* Look for public rules matching rulename. */
772  for (itor = hash_table_iter(imp->rules); itor;
773  itor = hash_table_iter_next(itor)) {
774  hash_entry_t *he = itor->ent;
775  jsgf_rule_t *rule = hash_entry_val(he);
776  int rule_matches;
777  char *rule_name = importname2rulename(name);
778 
779  if (import_all) {
780  /* Match package name (symbol table is shared) */
781  rule_matches = !strncmp(rule_name, rule->name, packlen + 1);
782  }
783  else {
784  /* Exact match */
785  rule_matches = !strcmp(rule_name, rule->name);
786  }
787  ckd_free(rule_name);
788  if (rule->is_public && rule_matches) {
789  void *val;
790  char *newname;
791 
792  /* Link this rule into the current namespace. */
793  c = strrchr(rule->name, '.');
794  assert(c != NULL);
795  newname = jsgf_fullname(jsgf, c);
796 
797  E_INFO("Imported %s\n", newname);
798  val = hash_table_enter(jsgf->rules, newname,
799  jsgf_rule_retain(rule));
800  if (val != (void *)rule) {
801  E_WARN("Multiply defined symbol: %s\n", newname);
802  }
803  if (!import_all) {
804  hash_table_iter_free(itor);
805  return rule;
806  }
807  }
808  }
809  }
810 
811  return NULL;
812 }
