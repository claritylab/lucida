68 int
69 fe_parse_general_params(cmd_ln_t *config, fe_t * fe)
70 {
71  int j, frate;
72 
73  fe->config = config;
74  fe->sampling_rate = cmd_ln_float32_r(config, "-samprate");
75  frate = cmd_ln_int32_r(config, "-frate");
76  if (frate > MAX_INT16 || frate > fe->sampling_rate || frate < 1) {
77  E_ERROR
78  ("Frame rate %d can not be bigger than sample rate %.02f\n",
79  frate, fe->sampling_rate);
80  return -1;
81  }
82 
83  fe->frame_rate = (int16)frate;
84  if (cmd_ln_boolean_r(config, "-dither")) {
85  fe->dither = 1;
86  fe->seed = cmd_ln_int32_r(config, "-seed");
87  }
88 #ifdef WORDS_BIGENDIAN
89  fe->swap = strcmp("big", cmd_ln_str_r(config, "-input_endian")) == 0 ? 0 : 1;
90 #else
91  fe->swap = strcmp("little", cmd_ln_str_r(config, "-input_endian")) == 0 ? 0 : 1;
92 #endif
93  fe->window_length = cmd_ln_float32_r(config, "-wlen");
94  fe->pre_emphasis_alpha = cmd_ln_float32_r(config, "-alpha");
95 
96  fe->num_cepstra = (uint8)cmd_ln_int32_r(config, "-ncep");
97  fe->fft_size = (int16)cmd_ln_int32_r(config, "-nfft");
98 
99  /* Check FFT size, compute FFT order (log_2(n)) */
100  for (j = fe->fft_size, fe->fft_order = 0; j > 1; j >>= 1, fe->fft_order++) {
101  if (((j % 2) != 0) || (fe->fft_size <= 0)) {
102  E_ERROR("fft: number of points must be a power of 2 (is %d)\n",
103  fe->fft_size);
104  return -1;
105  }
106  }
107  /* Verify that FFT size is greater or equal to window length. */
108  if (fe->fft_size < (int)(fe->window_length * fe->sampling_rate)) {
109  E_ERROR("FFT: Number of points must be greater or equal to frame size (%d samples)\n",
110  (int)(fe->window_length * fe->sampling_rate));
111  return -1;
112  }
113 
114  fe->prespch_len = (int16)cmd_ln_int32_r(config, "-vad_prespeech");
115  fe->postspch_len = (int16)cmd_ln_int32_r(config, "-vad_postspeech");
116 
117  fe->remove_dc = cmd_ln_boolean_r(config, "-remove_dc");
118  fe->remove_noise = cmd_ln_boolean_r(config, "-remove_noise");
119  fe->remove_silence = cmd_ln_boolean_r(config, "-remove_silence");
120 
121  if (0 == strcmp(cmd_ln_str_r(config, "-transform"), "dct"))
122  fe->transform = DCT_II;
123  else if (0 == strcmp(cmd_ln_str_r(config, "-transform"), "legacy"))
124  fe->transform = LEGACY_DCT;
125  else if (0 == strcmp(cmd_ln_str_r(config, "-transform"), "htk"))
126  fe->transform = DCT_HTK;
127  else {
128  E_ERROR("Invalid transform type (values are 'dct', 'legacy', 'htk')\n");
129  return -1;
130  }
131 
132  if (cmd_ln_boolean_r(config, "-logspec"))
133  fe->log_spec = RAW_LOG_SPEC;
134  if (cmd_ln_boolean_r(config, "-smoothspec"))
135  fe->log_spec = SMOOTH_LOG_SPEC;
136 
137  return 0;
138 }

156  mel->doublewide = cmd_ln_boolean_r(config, "-doublebw");
157 
158  mel->warp_type = cmd_ln_str_r(config, "-warp_type");
159  mel->warp_params = cmd_ln_str_r(config, "-warp_params");
160  mel->lifter_val = cmd_ln_int32_r(config, "-lifter");
161

163  mel->round_filters = cmd_ln_boolean_r(config, "-round_filters");
164

166  E_ERROR("Failed to initialize the warping function.\n");

202  fe->mel_fb->unit_area ? "" : "not ");
203  E_INFO("Will %sround filter frequencies to DFT points\n",
204  fe->mel_fb->round_filters ? "" : "not ");
205  E_INFO("Will %suse double bandwidth in mel filter\n",
206  fe->mel_fb->doublewide ? "" : "not ");
207 }

215 fe_t *
216 fe_init_auto_r(cmd_ln_t *config)
217 {
218  fe_t *fe;
219  int prespch_frame_len;
220 
221  fe = (fe_t*)ckd_calloc(1, sizeof(*fe));
222  fe->refcount = 1;
223 
224  /* transfer params to front end */
225  if (fe_parse_general_params(cmd_ln_retain(config), fe) < 0) {
226  fe_free(fe);
227  return NULL;
228  }
229 
230  /* compute remaining fe parameters */
231  /* We add 0.5 so approximate the float with the closest
232  * integer. E.g., 2.3 is truncate to 2, whereas 3.7 becomes 4
233  */
234  fe->frame_shift = (int32) (fe->sampling_rate / fe->frame_rate + 0.5);
235  fe->frame_size = (int32) (fe->window_length * fe->sampling_rate + 0.5);
236  fe->prior = 0;
237  fe->frame_counter = 0;
238 
239  assert (fe->frame_shift > 1);
240 
241  if (fe->frame_size > (fe->fft_size)) {
242  E_WARN
243  ("Number of FFT points has to be a power of 2 higher than %d\n",
244  (fe->frame_size));
245  fe_free(fe);
246  return (NULL);
247  }
248 
249  if (fe->dither)
250  fe_init_dither(fe->seed);
251 
252  /* establish buffers for overflow samps and hamming window */
253  fe->overflow_samps = ckd_calloc(fe->frame_size, sizeof(int16));
254  fe->hamming_window = ckd_calloc(fe->frame_size/2, sizeof(window_t));
255 
256  /* create hamming window */
257  fe_create_hamming(fe->hamming_window, fe->frame_size);
258 
259  /* init and fill appropriate filter structure */
260  fe->mel_fb = ckd_calloc(1, sizeof(*fe->mel_fb));
261 
262  /* transfer params to mel fb */
263  fe_parse_melfb_params(config, fe, fe->mel_fb);
264  fe_build_melfilters(fe->mel_fb);
265 
266  fe_compute_melcosine(fe->mel_fb);
267  if (fe->remove_noise || fe->remove_silence)
268  fe->noise_stats = fe_init_noisestats(fe->mel_fb->num_filters);
269 
270  fe->vad_data = (vad_data_t*)ckd_calloc(1, sizeof(*fe->vad_data));
271  prespch_frame_len = fe->log_spec != RAW_LOG_SPEC ? fe->num_cepstra : fe->mel_fb->num_filters;
272  fe->vad_data->prespch_buf = fe_init_prespch(fe->prespch_len + 1, prespch_frame_len, fe->frame_shift);
273 
274  /* Create temporary FFT, spectrum and mel-spectrum buffers. */
275  /* FIXME: Gosh there are a lot of these. */
276  fe->spch = ckd_calloc(fe->frame_size, sizeof(*fe->spch));
277  fe->frame = ckd_calloc(fe->fft_size, sizeof(*fe->frame));
278  fe->spec = ckd_calloc(fe->fft_size, sizeof(*fe->spec));
279  fe->mfspec = ckd_calloc(fe->mel_fb->num_filters, sizeof(*fe->mfspec));
280 
281  /* create twiddle factors */
282  fe->ccc = ckd_calloc(fe->fft_size / 4, sizeof(*fe->ccc));
283  fe->sss = ckd_calloc(fe->fft_size / 4, sizeof(*fe->sss));
284  fe_create_twiddle(fe);
285 
286  if (cmd_ln_boolean_r(config, "-verbose")) {
287  fe_print_current(fe);
288  }
289 
290  /*** Z.A.B. ***/
291  /*** Initialize the overflow buffers ***/
292  fe_start_utt(fe);
293  return fe;
294 }

373 int
374 fe_process_frames(fe_t *fe,
375  int16 const **inout_spch,
376  size_t *inout_nsamps,
377  mfcc_t **buf_cep,
378  int32 *inout_nframes)
379 {
380  int32 frame_count;
381  int outidx, i, n_overflow, orig_n_overflow;
382  int16 const *orig_spch;
383 
384  /* In the special case where there is no output buffer, return the
385  * maximum number of frames which would be generated. */
386  if (buf_cep == NULL) {
387  if (*inout_nsamps + fe->num_overflow_samps < (size_t)fe->frame_size)
388  *inout_nframes = 0;
389  else
390  *inout_nframes = 1
391  + ((*inout_nsamps + fe->num_overflow_samps - fe->frame_size)
392  / fe->frame_shift);
393  return *inout_nframes;
394  }
395 
396  /* Are there not enough samples to make at least 1 frame? */
397  if (*inout_nsamps + fe->num_overflow_samps < (size_t)fe->frame_size) {
398  if (*inout_nsamps > 0) {
399  /* Append them to the overflow buffer. */
400  memcpy(fe->overflow_samps + fe->num_overflow_samps,
401  *inout_spch, *inout_nsamps * (sizeof(int16)));
402  fe->num_overflow_samps += *inout_nsamps;
403  /* Update input-output pointers and counters. */
404  *inout_spch += *inout_nsamps;
405  *inout_nsamps = 0;
406  }
407  /* We produced no frames of output, sorry! */
408  *inout_nframes = 0;
409  return 0;
410  }
411 
412  /* Can't write a frame? Then do nothing! */
413  if (*inout_nframes < 1) {
414  *inout_nframes = 0;
415  return 0;
416  }
417 
418  /* Index of output frame. */
419  outidx = 0;
420 
421  /* Try to read from prespeech buffer */
422  if (fe->vad_data->global_state) {
423  while ((*inout_nframes) > 0 && fe_prespch_read_cep(fe->vad_data->prespch_buf, buf_cep[outidx]) > 0) {
424  outidx++;
425  (*inout_nframes)--;
426  }
427  if ((*inout_nframes) < 1) {
428  /* mfcc buffer is filled from prespeech buffer */
429  *inout_nframes = outidx;
430  return 0;
431  }
432  }
433 
434  /* Keep track of the original start of the buffer. */
435  orig_spch = *inout_spch;
436  orig_n_overflow = fe->num_overflow_samps;
437  /* How many frames will we be able to get? */
438  frame_count = 1
439  + ((*inout_nsamps + fe->num_overflow_samps - fe->frame_size)
440  / fe->frame_shift);
441  /* Limit it to the number of output frames available. */
442  if (frame_count > *inout_nframes)
443  frame_count = *inout_nframes;
444 
445  /* Start processing, taking care of any incoming overflow. */
446  if (fe->num_overflow_samps) {
447  int offset = fe->frame_size - fe->num_overflow_samps;
448 
449  /* Append start of spch to overflow samples to make a full frame. */
450  memcpy(fe->overflow_samps + fe->num_overflow_samps,
451  *inout_spch, offset * sizeof(**inout_spch));
452  fe_read_frame(fe, fe->overflow_samps, fe->frame_size);
453  /* Update input-output pointers and counters. */
454  *inout_spch += offset;
455  *inout_nsamps -= offset;
456  fe->num_overflow_samps -= fe->frame_shift;
457  } else {
458  fe_read_frame(fe, *inout_spch, fe->frame_size);
459  /* Update input-output pointers and counters. */
460  *inout_spch += fe->frame_size;
461  *inout_nsamps -= fe->frame_size;
462  }
463  fe_write_frame(fe, buf_cep[outidx]);
464  if (!fe->vad_data->state_changed && fe->vad_data->global_state)
465  outidx++;
466 
467  if (fe->vad_data->state_changed && fe->vad_data->global_state) {
468  /* previous frame triggered vad into speech state
469  * dumping prespeech buffer */
470  while ((*inout_nframes) > 0 && fe_prespch_read_cep(fe->vad_data->prespch_buf, buf_cep[outidx]) > 0) {
471  outidx++;
472  (*inout_nframes)--;
473  frame_count--;
474  }
475  }
476 
477  /* Process all remaining frames. */
478  for (i = 1; i < frame_count; ++i) {
479  assert(*inout_nsamps >= (size_t)fe->frame_shift);
480  fe_shift_frame(fe, *inout_spch, fe->frame_shift);
481  fe_write_frame(fe, buf_cep[outidx]);
482  if (!fe->vad_data->state_changed && fe->vad_data->global_state)
483  outidx++;
484  /* Update input-output pointers and counters. */
485  *inout_spch += fe->frame_shift;
486  *inout_nsamps -= fe->frame_shift;
487  /* Amount of data behind the original input which is still needed. */
488  if (fe->num_overflow_samps > 0)
489  fe->num_overflow_samps -= fe->frame_shift;
490 
491  if (fe->vad_data->state_changed && fe->vad_data->global_state) {
492  /* previous frame triggered vad into speech state */
493  while (i < frame_count && fe_prespch_read_cep(fe->vad_data->prespch_buf, buf_cep[outidx]) != 0) {
494  i++;
495  outidx++;
496  }
497  }
498  }
499 
500  /* How many relevant overflow samples are there left? */
501  if (fe->num_overflow_samps <= 0) {
502  /* Maximum number of overflow samples past *inout_spch to save. */
503  n_overflow = *inout_nsamps;
504  if (n_overflow > fe->frame_shift)
505  n_overflow = fe->frame_shift;
506  fe->num_overflow_samps = fe->frame_size - fe->frame_shift;
507  /* Make sure this isn't an illegal read! */
508  if (fe->num_overflow_samps > *inout_spch - orig_spch)
509  fe->num_overflow_samps = *inout_spch - orig_spch;
510  fe->num_overflow_samps += n_overflow;
511  if (fe->num_overflow_samps > 0) {
512  memcpy(fe->overflow_samps,
513  *inout_spch - (fe->frame_size - fe->frame_shift),
514  fe->num_overflow_samps * sizeof(**inout_spch));
515  /* Update the input pointer to cover this stuff. */
516  *inout_spch += n_overflow;
517  *inout_nsamps -= n_overflow;
518  }
519  } else {
520  /* There is still some relevant data left in the overflow buffer. */
521  /* Shift existing data to the beginning. */
522  memmove(fe->overflow_samps,
523  fe->overflow_samps + orig_n_overflow - fe->num_overflow_samps,
524  fe->num_overflow_samps * sizeof(*fe->overflow_samps));
525  /* Copy in whatever we had in the original speech buffer. */
526  n_overflow = *inout_spch - orig_spch + *inout_nsamps;
527  if (n_overflow > fe->frame_size - fe->num_overflow_samps)
528  n_overflow = fe->frame_size - fe->num_overflow_samps;
529  memcpy(fe->overflow_samps + fe->num_overflow_samps,
530  orig_spch, n_overflow * sizeof(*orig_spch));
531  fe->num_overflow_samps += n_overflow;
532  /* Advance the input pointers. */
533  if (n_overflow > *inout_spch - orig_spch) {
534  n_overflow -= (*inout_spch - orig_spch);
535  *inout_spch += n_overflow;
536  *inout_nsamps -= n_overflow;
537  }
538  }
539 
540  /* Finally update the frame counter with the number of frames we procesed. */
541  *inout_nframes = outidx; /* FIXME: Not sure why I wrote it this way... */
542  return 0;
543 }

547  int16 const **inout_spch,
