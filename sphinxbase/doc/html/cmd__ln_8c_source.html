106 /*
107  * Find max length of name and default fields in the given defn array.
108  * Return #items in defn array.
109  */
110 static int32
111 arg_strlen(const arg_t * defn, int32 * namelen, int32 * deflen)
112 {
113  int32 i, l;
114 
115  *namelen = *deflen = 0;
116  for (i = 0; defn[i].name; i++) {
117  l = strlen(defn[i].name);
118  if (*namelen < l)
119  *namelen = l;
120 
121  if (defn[i].deflt)
122  l = strlen(defn[i].deflt);
123  else
124  l = strlen("(null)");
125  /* E_INFO("string default, %s , name %s, length %d\n",defn[i].deflt,defn[i].name,l); */
126  if (*deflen < l)
127  *deflen = l;
128  }
129 
130  return i;
131 }

142 static const arg_t **
143 arg_sort(const arg_t * defn, int32 n)
144 {
145  const arg_t ** pos;
146  int32 i;
147 
148  pos = (const arg_t **) ckd_calloc(n, sizeof(arg_t *));
149  for (i = 0; i < n; ++i)

151  qsort(pos, n, sizeof(arg_t *), cmp_name);

156 static size_t
157 strnappend(char **dest, size_t *dest_allocation,
158  const char *source, size_t n)
159 {
160  size_t source_len, required_allocation;
161 
162  if (dest == NULL || dest_allocation == NULL)
163  return -1;
164  if (*dest == NULL && *dest_allocation != 0)
165  return -1;
166  if (source == NULL)
167  return *dest_allocation;
168 
169  source_len = strlen(source);
170  if (n && n < source_len)
171  source_len = n;
172 
173  required_allocation = (*dest ? strlen(*dest) : 0) + source_len + 1;
174  if (*dest_allocation < required_allocation) {
175  if (*dest_allocation == 0) {
176  *dest = ckd_calloc(required_allocation * 2, 1);
177  } else {
178  *dest = ckd_realloc(*dest, required_allocation * 2);
179  }
180  *dest_allocation = required_allocation * 2;
181  }
182 
183  strncat(*dest, source, source_len);
184 
185  return *dest_allocation;
186 }

195 static char*
196 arg_resolve_env(const char *str)
197 {
198  char *resolved_str = NULL;
199  char env_name[100];
200  const char *env_val;
201  size_t alloced = 0;
202  const char *i = str, *j;
203 
204  /* calculate required resolved_str size */
205  do {
206  j = strstr(i, "$(");
207  if (j != NULL) {
208  if (j != i) {
209  strnappend(&resolved_str, &alloced, i, j - i);
210  i = j;
211  }
212  j = strchr(i + 2, ')');
213  if (j != NULL) {
214  if (j - (i + 2) < 100) {
215  strncpy(env_name, i + 2, j - (i + 2));
216  env_name[j - (i + 2)] = '\0';
217  #if !defined(_WIN32_WCE)
218  env_val = getenv(env_name);
219  if (env_val)
220  strappend(&resolved_str, &alloced, env_val);
221  #else
222  env_val = 0;
223  #endif
224  }
225  i = j + 1;
226  } else {
227  /* unclosed, copy and skip */
228  j = i + 2;
229  strnappend(&resolved_str, &alloced, i, j - i);
230  i = j;
231  }
232  } else {
233  strappend(&resolved_str, &alloced, i);
234  }
235  } while(j != NULL);
236 
237  return resolved_str;
238 }

240 static void
241 arg_dump_r(cmd_ln_t *cmdln, FILE *fp, const arg_t * defn, int32 doc)
242 {
243  const arg_t **pos;
244  int32 i, l, n;
245  int32 namelen, deflen;
246  anytype_t *vp;
247  char const **array;
248 
249  /* No definitions, do nothing. */
250  if (defn == NULL || fp == NULL)
251  return;
252 
253  /* Find max lengths of name and default value fields, and #entries in defn */
254  n = arg_strlen(defn, &namelen, &deflen);
255  /* E_INFO("String length %d. Name length %d, Default Length %d\n",n, namelen, deflen); */
256  namelen = namelen & 0xfffffff8; /* Previous tab position */
257  deflen = deflen & 0xfffffff8; /* Previous tab position */
258 
259  fprintf(fp, "[NAME]");
260  for (l = strlen("[NAME]"); l < namelen; l += 8)
261  fprintf(fp, "\t");
262  fprintf(fp, "\t[DEFLT]");
263  for (l = strlen("[DEFLT]"); l < deflen; l += 8)
264  fprintf(fp, "\t");
265 
266  if (doc) {
267  fprintf(fp, "\t[DESCR]\n");
268  }
269  else {
270  fprintf(fp, "\t[VALUE]\n");
271  }
272 
273  /* Print current configuration, sorted by name */
274  pos = arg_sort(defn, n);
275  for (i = 0; i < n; i++) {
276  fprintf(fp, "%s", pos[i]->name);
277  for (l = strlen(pos[i]->name); l < namelen; l += 8)
278  fprintf(fp, "\t");
279 
280  fprintf(fp, "\t");
281  if (pos[i]->deflt) {
282  fprintf(fp, "%s", pos[i]->deflt);
283  l = strlen(pos[i]->deflt);
284  }
285  else
286  l = 0;
287  for (; l < deflen; l += 8)
288  fprintf(fp, "\t");
289 
290  fprintf(fp, "\t");
291  if (doc) {
292  if (pos[i]->doc)
293  fprintf(fp, "%s", pos[i]->doc);
294  }
295  else {
296  vp = cmd_ln_access_r(cmdln, pos[i]->name);
297  if (vp) {
298  switch (pos[i]->type) {
299  case ARG_INTEGER:
300  case REQARG_INTEGER:
301  fprintf(fp, "%ld", vp->i);
302  break;
303  case ARG_FLOATING:
304  case REQARG_FLOATING:
305  fprintf(fp, "%e", vp->fl);
306  break;
307  case ARG_STRING:
308  case REQARG_STRING:
309  if (vp->ptr)
310  fprintf(fp, "%s", (char *)vp->ptr);
311  break;
312  case ARG_STRING_LIST:
313  array = (char const**)vp->ptr;
314  if (array)
315  for (l = 0; array[l] != 0; l++) {
316  fprintf(fp, "%s,", array[l]);
317  }
318  break;
319  case ARG_BOOLEAN:
320  case REQARG_BOOLEAN:
321  fprintf(fp, "%s", vp->i ? "yes" : "no");
322  break;
323  default:
324  E_ERROR("Unknown argument type: %d\n", pos[i]->type);
325  }
326  }
327  }
328 
329  fprintf(fp, "\n");
330  }
331  ckd_free(pos);
332 
333  fprintf(fp, "\n");
334 }

336 static char **
337 parse_string_list(const char *str)
338 {
339  int count, i, j;
340  const char *p;
341  char **result;
342 
343  p = str;
344  count = 1;
345  while (*p) {
346  if (*p == ',')
347  count++;
348  p++;
349  }
350  /* Should end with NULL */
351  result = (char **) ckd_calloc(count + 1, sizeof(char *));
352  p = str;
353  for (i = 0; i < count; i++) {
354  for (j = 0; p[j] != ',' && p[j] != 0; j++);
355  result[i] = ckd_calloc(j + 1, sizeof(char));
356  strncpy( result[i], p, j);
357  p = p + j + 1;
358  }
359  return result;
360 }

362 static cmd_ln_val_t *
363 cmd_ln_val_init(int t, const char *str)
364 {
365  cmd_ln_val_t *v;
366  anytype_t val;
367  char *e_str;
368 
369  if (!str) {
370  /* For lack of a better default value. */
371  memset(&val, 0, sizeof(val));
372  }
373  else {
374  int valid = 1;
375  e_str = arg_resolve_env(str);
376 
377  switch (t) {
378  case ARG_INTEGER:
379  case REQARG_INTEGER:
380  if (sscanf(e_str, "%ld", &val.i) != 1)
381  valid = 0;
382  break;
383  case ARG_FLOATING:
384  case REQARG_FLOATING:
385  if (e_str == NULL || e_str[0] == 0)
386  valid = 0;
387  val.fl = atof_c(e_str);
388  break;
389  case ARG_BOOLEAN:
390  case REQARG_BOOLEAN:
391  if ((e_str[0] == 'y') || (e_str[0] == 't') ||
392  (e_str[0] == 'Y') || (e_str[0] == 'T') || (e_str[0] == '1')) {
393  val.i = TRUE;
394  }
395  else if ((e_str[0] == 'n') || (e_str[0] == 'f') ||
396  (e_str[0] == 'N') || (e_str[0] == 'F') |
397  (e_str[0] == '0')) {
398  val.i = FALSE;
399  }
400  else {
401  E_ERROR("Unparsed boolean value '%s'\n", str);
402  valid = 0;
403  }
404  break;
405  case ARG_STRING:
406  case REQARG_STRING:
407  val.ptr = ckd_salloc(e_str);
408  break;
409  case ARG_STRING_LIST:
410  val.ptr = parse_string_list(e_str);
411  break;
412  default:
413  E_ERROR("Unknown argument type: %d\n", t);
414  valid = 0;
415  }
416 
417  ckd_free(e_str);
418  if (valid == 0)
419  return NULL;
420  }
421 
422  v = ckd_calloc(1, sizeof(*v));
423  memcpy(v, &val, sizeof(val));
424  v->type = t;
425 
426  return v;
427 }

429 /*
430  * Handles option parsing for cmd_ln_parse_file_r() and cmd_ln_init()
431  * also takes care of storing argv.
432  * DO NOT call it from cmd_ln_parse_r()
433  */
434 static cmd_ln_t *
435 parse_options(cmd_ln_t *cmdln, const arg_t *defn, int32 argc, char* argv[], int32 strict)
436 {
437  cmd_ln_t *new_cmdln;
438 
439  new_cmdln = cmd_ln_parse_r(cmdln, defn, argc, argv, strict);
440  /* If this failed then clean up and return NULL. */
441  if (new_cmdln == NULL) {
442  int32 i;
443  for (i = 0; i < argc; ++i)
444  ckd_free(argv[i]);
445  ckd_free(argv);
446  return NULL;
447  }
448 
449  /* Otherwise, we need to add the contents of f_argv to the new object. */
450  if (new_cmdln == cmdln) {
451  /* If we are adding to a previously passed-in cmdln, then
452  * store our allocated strings in its f_argv. */
453  new_cmdln->f_argv = ckd_realloc(new_cmdln->f_argv,
454  (new_cmdln->f_argc + argc)
455  * sizeof(*new_cmdln->f_argv));
456  memcpy(new_cmdln->f_argv + new_cmdln->f_argc, argv,
457  argc * sizeof(*argv));
458  ckd_free(argv);
459  new_cmdln->f_argc += argc;
460  }
461  else {
462  /* Otherwise, store f_argc and f_argv. */
463  new_cmdln->f_argc = argc;
464  new_cmdln->f_argv = argv;
465  }
466 
467  return new_cmdln;
468 }

494 void
495 cmd_ln_appl_enter(int argc, char *argv[],
496  const char *default_argfn,
497  const arg_t * defn)
498 {
499  /* Look for default or specified arguments file */
500  const char *str;
501 
502  str = NULL;
503 
504  if ((argc == 2) && (strcmp(argv[1], "help") == 0)) {
505  cmd_ln_print_help(stderr, defn);
506  exit(1);
507  }
508 
509  if ((argc == 2) && (argv[1][0] != '-'))
510  str = argv[1];
511  else if (argc == 1) {
512  FILE *fp;
513  E_INFO("Looking for default argument file: %s\n", default_argfn);
514 
515  if ((fp = fopen(default_argfn, "r")) == NULL) {
516  E_INFO("Can't find default argument file %s.\n",
517  default_argfn);
518  }
519  else {
520  str = default_argfn;
521  }
522  if (fp != NULL)
523  fclose(fp);
524  }
525 
526 
527  if (str) {
528  /* Build command line argument list from file */
529  E_INFO("Parsing command lines from file %s\n", str);
530  if (cmd_ln_parse_file(defn, str, TRUE)) {
531  E_INFOCONT("Usage:\n");
532  E_INFOCONT("\t%s argument-list, or\n", argv[0]);
533  E_INFOCONT("\t%s [argument-file] (default file: . %s)\n\n",
534  argv[0], default_argfn);
535  cmd_ln_print_help(stderr, defn);
536  exit(1);
537  }
538  }
539  else {
540  cmd_ln_parse(defn, argc, argv, TRUE);
541  }
542 }

551 cmd_ln_t *
552 cmd_ln_parse_r(cmd_ln_t *inout_cmdln, const arg_t * defn, int32 argc, char *argv[], int strict)
553 {
554  int32 i, j, n, argstart;
555  hash_table_t *defidx = NULL;
556  cmd_ln_t *cmdln;
557 
558  /* Construct command-line object */
559  if (inout_cmdln == NULL) {
560  cmdln = ckd_calloc(1, sizeof(*cmdln));
561  cmdln->refcount = 1;
562  }
563  else
564  cmdln = inout_cmdln;
565 
566  /* Build a hash table for argument definitions */
567  defidx = hash_table_new(50, 0);
568  if (defn) {
569  for (n = 0; defn[n].name; n++) {
570  void *v;
571 
572  v = hash_table_enter(defidx, defn[n].name, (void *)&defn[n]);
573  if (strict && (v != &defn[n])) {
574  E_ERROR("Duplicate argument name in definition: %s\n", defn[n].name);
575  goto error;
576  }
577  }
578  }
579  else {
580  /* No definitions. */
581  n = 0;
582  }
583 
584  /* Allocate memory for argument values */
585  if (cmdln->ht == NULL)
586  cmdln->ht = hash_table_new(n, 0 /* argument names are case-sensitive */ );
587 
588 
589  /* skip argv[0] if it doesn't start with dash */
590  argstart = 0;
591  if (argc > 0 && argv[0][0] != '-') {
592  argstart = 1;
593  }
594 
595  /* Parse command line arguments (name-value pairs) */
596  for (j = argstart; j < argc; j += 2) {
597  arg_t *argdef;
598  cmd_ln_val_t *val;
599  void *v;
600 
601  if (hash_table_lookup(defidx, argv[j], &v) < 0) {
602  if (strict) {
603  E_ERROR("Unknown argument name '%s'\n", argv[j]);
604  goto error;
605  }
606  else if (defn == NULL)
607  v = NULL;
608  else
609  continue;
610  }
611  argdef = v;
612 
613  /* Enter argument value */
614  if (j + 1 >= argc) {
615  cmd_ln_print_help_r(cmdln, stderr, defn);
616  E_ERROR("Argument value for '%s' missing\n", argv[j]);
617  goto error;
618  }
619 
620  if (argdef == NULL)
621  val = cmd_ln_val_init(ARG_STRING, argv[j + 1]);
622  else {
623  if ((val = cmd_ln_val_init(argdef->type, argv[j + 1])) == NULL) {
624  cmd_ln_print_help_r(cmdln, stderr, defn);
625  E_ERROR("Bad argument value for %s: %s\n", argv[j],
626  argv[j + 1]);
627  goto error;
628  }
629  }
630 
631  if ((v = hash_table_enter(cmdln->ht, argv[j], (void *)val)) !=
632  (void *)val)
633  {
634  if (strict) {
635  cmd_ln_val_free(val);
636  E_ERROR("Duplicate argument name in arguments: %s\n",
637  argdef->name);
638  goto error;
639  }
640  else {
641  v = hash_table_replace(cmdln->ht, argv[j], (void *)val);
642  cmd_ln_val_free((cmd_ln_val_t *)v);
643  }
644  }
645  }
646 
647  /* Fill in default values, if any, for unspecified arguments */
648  for (i = 0; i < n; i++) {
649  cmd_ln_val_t *val;
650  void *v;
651 
652  if (hash_table_lookup(cmdln->ht, defn[i].name, &v) < 0) {
653  if ((val = cmd_ln_val_init(defn[i].type, defn[i].deflt)) == NULL) {
654  E_ERROR
655  ("Bad default argument value for %s: %s\n",
656  defn[i].name, defn[i].deflt);
657  goto error;
658  }
659  hash_table_enter(cmdln->ht, defn[i].name, (void *)val);
660  }
661  }
662 
663  /* Check for required arguments; exit if any missing */
664  j = 0;
665  for (i = 0; i < n; i++) {
666  if (defn[i].type & ARG_REQUIRED) {
667  void *v;
668  if (hash_table_lookup(cmdln->ht, defn[i].name, &v) != 0)
669  E_ERROR("Missing required argument %s\n", defn[i].name);
670  }
671  }
672  if (j > 0) {
673  cmd_ln_print_help_r(cmdln, stderr, defn);
674  goto error;
675  }
676 
677  if (strict && argc == 1) {
678  E_ERROR("No arguments given, available options are:\n");
679  cmd_ln_print_help_r(cmdln, stderr, defn);
680  if (defidx)
681  hash_table_free(defidx);
682  if (inout_cmdln == NULL)
683  cmd_ln_free_r(cmdln);
684  return NULL;
685  }
686 
687 #ifndef _WIN32_WCE
688  /* Set up logging. We need to do this earlier because we want to dump
689  * the information to the configured log, not to the stderr. */
690  if (cmd_ln_exists_r(cmdln, "-logfn") && cmd_ln_str_r(cmdln, "-logfn")) {
691  if (err_set_logfile(cmd_ln_str_r(cmdln, "-logfn")) < 0)
692  E_FATAL_SYSTEM("cannot redirect log output");
693  }
694 
695  /* Echo command line */
696  E_INFO("Parsing command line:\n");
697  for (i = 0; i < argc; i++) {
698  if (argv[i][0] == '-')

776 cmd_ln_t *
777 cmd_ln_parse_file_r(cmd_ln_t *inout_cmdln, const arg_t * defn, const char *filename, int32 strict)
778 {
779  FILE *file;
780  int argc;
781  int argv_size;
782  char *str;
783  int arg_max_length = 512;
784  int len = 0;
785  int quoting, ch;
786  char **f_argv;
787  int rv = 0;
788  const char separator[] = " \t\r\n";
789 
790  if ((file = fopen(filename, "r")) == NULL) {
791  E_ERROR("Cannot open configuration file %s for reading\n",
792  filename);
793  return NULL;
794  }
795 
796  ch = fgetc(file);
797  /* Skip to the next interesting character */
798  for (; ch != EOF && strchr(separator, ch); ch = fgetc(file)) ;
799 
800  if (ch == EOF) {
801  fclose(file);
802  return NULL;
803  }
804 
805  /*
806  * Initialize default argv, argc, and argv_size.
807  */
808  argv_size = 10;
809  argc = 0;
810  f_argv = ckd_calloc(argv_size, sizeof(char *));
811  /* Silently make room for \0 */
812  str = ckd_calloc(arg_max_length + 1, sizeof(char));
813  quoting = 0;
814 
815  do {
816  /* Handle arguments that are commented out */
817  if (len == 0 && argc % 2 == 0) {
818  while (ch == '#') {
819  /* Skip everything until newline */
820  for (ch = fgetc(file); ch != EOF && ch != '\n'; ch = fgetc(file)) ;
821  /* Skip to the next interesting character */
822  for (ch = fgetc(file); ch != EOF && strchr(separator, ch); ch = fgetc(file)) ;
823  }
824 
825  /* Check if we are at the last line (without anything interesting in it) */
826  if (ch == EOF)
827  break;
828  }
829 
830  /* Handle quoted arguments */
831  if (ch == '"' || ch == '\'') {
832  if (quoting == ch) /* End a quoted section with the same type */
833  quoting = 0;
834  else if (quoting) {
835  E_ERROR("Nesting quotations is not supported!\n");
836  rv = 1;
837  break;
838  }
839  else
840  quoting = ch; /* Start a quoted section */
841  }
842  else if (ch == EOF || (!quoting && strchr(separator, ch))) {
843  /* Reallocate argv so it is big enough to contain all the arguments */
844  if (argc >= argv_size) {
845  char **tmp_argv;
846  if (!(tmp_argv =
847  ckd_realloc(f_argv, argv_size * 2 * sizeof(char *)))) {
848  rv = 1;
849  break;
850  }
851  f_argv = tmp_argv;
852  argv_size *= 2;
853  }
854  /* Add the string to the list of arguments */
855  f_argv[argc] = ckd_salloc(str);
856  len = 0;
857  str[0] = '\0';
858  argc++;
859 
860  if (quoting)
861  E_WARN("Unclosed quotation, having EOF close it...\n");
862 
863  /* Skip to the next interesting character */
864  for (; ch != EOF && strchr(separator, ch); ch = fgetc(file)) ;
865 
866  if (ch == EOF)
867  break;
868 
869  /* We already have the next character */
870  continue;
871  }
872  else {
873  if (len >= arg_max_length) {
874  /* Make room for more chars (including the \0 !) */
875  char *tmp_str = str;
876  if ((tmp_str = ckd_realloc(str, (1 + arg_max_length * 2) * sizeof(char))) == NULL) {
877  rv = 1;
878  break;
879  }
880  str = tmp_str;
881  arg_max_length *= 2;
882  }
883  /* Add the char to the argument string */
884  str[len++] = ch;
885  /* Always null terminate */
886  str[len] = '\0';
887  }
888 
889  ch = fgetc(file);
890  } while (1);
891 
892  fclose(file);
893 
894  ckd_free(str);
895 
896  if (rv) {
897  for (ch = 0; ch < argc; ++ch)
898  ckd_free(f_argv[ch]);
899  ckd_free(f_argv);
900  return NULL;
901  }
902 
903  return parse_options(inout_cmdln, defn, argc, f_argv, strict);
904 }
