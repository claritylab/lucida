94 strcmp_nocase(const char *str1, const char *str2)
95 {
96  char c1, c2;
97 
98  if (str1 == str2)
99  return 0;
100  if (str1 && str2) {

102  c1 = *(str1++);
103  c1 = UPPER_CASE(c1);
104  c2 = *(str2++);
105  c2 = UPPER_CASE(c2);
106  if (c1 != c2)
107  return (c1 - c2);
108  if (c1 == '\0')
