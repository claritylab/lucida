81 void
82 err_msg(err_lvl_t lvl, const char *path, long ln, const char *fmt, ...)
83 {
84  static const char *err_prefix[ERR_MAX] = {
85  "DEBUG", "INFO", "INFOCONT", "WARN", "ERROR", "FATAL"
86  };
87 
88  char msg[1024];
89  va_list ap;
90 
91  if (!err_cb)
92  return;
93 
94  va_start(ap, fmt);
95  vsnprintf(msg, sizeof(msg), fmt, ap);
96  va_end(ap);
97 
98  if (path) {
99  const char *fname = path2basename(path);
100  if (lvl == ERR_INFOCONT)
101  err_cb(err_user_data, lvl, "%s(%ld): %s", fname, ln, msg);
102  else if (lvl == ERR_INFO)
103  err_cb(err_user_data, lvl, "%s: %s(%ld): %s", err_prefix[lvl], fname, ln, msg);
104  else
105  err_cb(err_user_data, lvl, "%s: \"%s\", line %ld: %s", err_prefix[lvl], fname, ln, msg);
106  } else {
107  err_cb(err_user_data, lvl, "%s", msg);
108  }
109 }

159 int
160 err_set_logfile(const char *path)
161 {
162  FILE *newfp, *oldfp;
163 
164  if ((newfp = fopen(path, "a")) == NULL)
165  return -1;
166  oldfp = err_get_logfp();
167  err_set_logfp(newfp);
168  if (oldfp != NULL && oldfp != stdout && oldfp != stderr)
169  fclose(oldfp);
170  return 0;
171 }
