76 struct noise_stats_s {
77  /* Smoothed power */
78  powspec_t *power;
79  /* Noise estimate */
80  powspec_t *noise;
81  /* Signal floor estimate */
82  powspec_t *floor;
83  /* Peak for temporal masking */
84  powspec_t *peak;
85 
86  /* Initialize it next time */
87  uint8 undefined;
88  /* Number of items to process */
89  uint32 num_filters;
90 
91  /* Precomputed constants */
92  powspec_t lambda_power;
93  powspec_t comp_lambda_power;
94  powspec_t lambda_a;
95  powspec_t comp_lambda_a;
96  powspec_t lambda_b;
97  powspec_t comp_lambda_b;
98  powspec_t lambda_t;
99  powspec_t mu_t;
100  powspec_t max_gain;
101  powspec_t inv_max_gain;
102 
103  powspec_t smooth_scaling[2 * SMOOTH_WINDOW + 3];
104 };

106 static void
107 fe_low_envelope(noise_stats_t *noise_stats, powspec_t * buf, powspec_t * floor_buf, int32 num_filt)
108 {
109  int i;
110 
111  for (i = 0; i < num_filt; i++) {
112 #ifndef FIXED_POINT
113  if (buf[i] >= floor_buf[i]) {
114  floor_buf[i] =
115  noise_stats->lambda_a * floor_buf[i] + noise_stats->comp_lambda_a * buf[i];
116  }
117  else {
118  floor_buf[i] =
119  noise_stats->lambda_b * floor_buf[i] + noise_stats->comp_lambda_b * buf[i];
120  }
121 #else
122  if (buf[i] >= floor_buf[i]) {
123  floor_buf[i] = fe_log_add(noise_stats->lambda_a + floor_buf[i],
124  noise_stats->comp_lambda_a + buf[i]);
125  }
126  else {
127  floor_buf[i] = fe_log_add(noise_stats->lambda_b + floor_buf[i],
128  noise_stats->comp_lambda_b + buf[i]);
129  }
130 #endif
131  }
132 }

134 /* temporal masking */
135 static void
136 fe_temp_masking(noise_stats_t *noise_stats, powspec_t * buf, powspec_t * peak, int32 num_filt)
137 {
138  powspec_t cur_in;
139  int i;
140 
141  for (i = 0; i < num_filt; i++) {
142  cur_in = buf[i];
143 
144 #ifndef FIXED_POINT
145  peak[i] *= noise_stats->lambda_t;
146  if (buf[i] < noise_stats->lambda_t * peak[i])
147  buf[i] = peak[i] * noise_stats->mu_t;
148 #else
149  peak[i] += noise_stats->lambda_t;
150  if (buf[i] < noise_stats->lambda_t + peak[i])
151  buf[i] = peak[i] + noise_stats->mu_t;
152 #endif
153 
154  if (cur_in > peak[i])
155  peak[i] = cur_in;
156  }
157 }

159 /* spectral weight smoothing */
160 static void
161 fe_weight_smooth(noise_stats_t *noise_stats, powspec_t * buf, powspec_t * coefs, int32 num_filt)
162 {
163  int i, j;
164  int l1, l2;
165  powspec_t coef;
166 
167  for (i = 0; i < num_filt; i++) {
168  l1 = ((i - SMOOTH_WINDOW) > 0) ? (i - SMOOTH_WINDOW) : 0;
169  l2 = ((i + SMOOTH_WINDOW) <
170  (num_filt - 1)) ? (i + SMOOTH_WINDOW) : (num_filt - 1);
171 
172 #ifndef FIXED_POINT
173  coef = 0;
174  for (j = l1; j <= l2; j++) {
175  coef += coefs[j];
176  }
177  buf[i] = buf[i] * (coef / (l2 - l1 + 1));
178 #else
179  coef = MIN_FIXLOG;
180  for (j = l1; j <= l2; j++) {
181  coef = fe_log_add(coef, coefs[j]);
182  }
183  buf[i] = buf[i] + coef - noise_stats->smooth_scaling[l2 - l1 + 1];
184 #endif
185 
186  }
187 }

189 noise_stats_t *
190 fe_init_noisestats(int num_filters)
191 {
192  int i;
193  noise_stats_t *noise_stats;
194 
195  noise_stats = (noise_stats_t *) ckd_calloc(1, sizeof(noise_stats_t));
196 
197  noise_stats->power =
198  (powspec_t *) ckd_calloc(num_filters, sizeof(powspec_t));
199  noise_stats->noise =
200  (powspec_t *) ckd_calloc(num_filters, sizeof(powspec_t));
201  noise_stats->floor =
202  (powspec_t *) ckd_calloc(num_filters, sizeof(powspec_t));
203  noise_stats->peak =
204  (powspec_t *) ckd_calloc(num_filters, sizeof(powspec_t));
205 
206  noise_stats->undefined = TRUE;
207  noise_stats->num_filters = num_filters;
208 
209 #ifndef FIXED_POINT
210  noise_stats->lambda_power = LAMBDA_POWER;
211  noise_stats->comp_lambda_power = 1 - LAMBDA_POWER;
212  noise_stats->lambda_a = LAMBDA_A;
213  noise_stats->comp_lambda_a = 1 - LAMBDA_A;
214  noise_stats->lambda_b = LAMBDA_B;
215  noise_stats->comp_lambda_b = 1 - LAMBDA_B;
216  noise_stats->lambda_t = LAMBDA_T;
217  noise_stats->mu_t = MU_T;
218  noise_stats->max_gain = MAX_GAIN;
219  noise_stats->inv_max_gain = 1.0 / MAX_GAIN;
220 
221  for (i = 0; i < 2 * SMOOTH_WINDOW + 1; i++) {
222  noise_stats->smooth_scaling[i] = 1.0 / i;
223  }
224 #else
225  noise_stats->lambda_power = FLOAT2FIX(log(LAMBDA_POWER));
226  noise_stats->comp_lambda_power = FLOAT2FIX(log(1 - LAMBDA_POWER));
227  noise_stats->lambda_a = FLOAT2FIX(log(LAMBDA_A));
228  noise_stats->comp_lambda_a = FLOAT2FIX(log(1 - LAMBDA_A));
229  noise_stats->lambda_b = FLOAT2FIX(log(LAMBDA_B));
230  noise_stats->comp_lambda_b = FLOAT2FIX(log(1 - LAMBDA_B));
231  noise_stats->lambda_t = FLOAT2FIX(log(LAMBDA_T));
232  noise_stats->mu_t = FLOAT2FIX(log(MU_T));
233  noise_stats->max_gain = FLOAT2FIX(log(MAX_GAIN));
234  noise_stats->inv_max_gain = FLOAT2FIX(log(1.0 / MAX_GAIN));
235 
236  for (i = 1; i < 2 * SMOOTH_WINDOW + 3; i++) {
237  noise_stats->smooth_scaling[i] = FLOAT2FIX(log(i));
238  }
239 #endif
240 
241  return noise_stats;
242 }

265 void
266 fe_track_snr(fe_t * fe)
267 {
268  powspec_t *signal;
269  powspec_t *gain;
270  noise_stats_t *noise_stats;
271  powspec_t *mfspec;
272  int32 i, num_filts;
273 
274  float lrt, snr;
275 
276  if (!(fe->remove_noise || fe->remove_silence)) {
277  //than nothing to do here,
278  //vad decision is always 1, to process everything
279  fe->vad_data->local_state = 1;
280  return;
281  }
282 
283  noise_stats = fe->noise_stats;
284  mfspec = fe->mfspec;
285  num_filts = noise_stats->num_filters;
286 
287  signal = (powspec_t *) ckd_calloc(num_filts, sizeof(powspec_t));
288 
289  if (noise_stats->undefined) {
290  for (i = 0; i < num_filts; i++) {
291  noise_stats->power[i] = mfspec[i];
292  noise_stats->noise[i] = mfspec[i];
293 #ifndef FIXED_POINT
294  noise_stats->floor[i] = mfspec[i] / noise_stats->max_gain;
295  noise_stats->peak[i] = 0.0;
296 #else
297  noise_stats->floor[i] = mfspec[i] - noise_stats->max_gain;
298  noise_stats->peak[i] = MIN_FIXLOG;
299 #endif
300  }
301  noise_stats->undefined = FALSE;
302  }
303 
304  /* Calculate smoothed power */
305  for (i = 0; i < num_filts; i++) {
306 #ifndef FIXED_POINT
307  noise_stats->power[i] =
308  noise_stats->lambda_power * noise_stats->power[i] + noise_stats->comp_lambda_power * mfspec[i];
309 #else
310  noise_stats->power[i] = fe_log_add(noise_stats->lambda_power + noise_stats->power[i],
311  noise_stats->comp_lambda_power + mfspec[i]);
312 #endif
313  }
314 
315  /* Noise estimation and vad decision */
316  fe_low_envelope(noise_stats, noise_stats->power, noise_stats->noise, num_filts);
317 
318  lrt = 0.0;
319  for (i = 0; i < num_filts; i++) {
320 #ifndef FIXED_POINT
321  signal[i] = noise_stats->power[i] - noise_stats->noise[i];
322  if (signal[i] < 0)
323  signal[i] = 0;
324  snr = log(noise_stats->power[i] / noise_stats->noise[i]);
325 #else
326  signal[i] = fe_log_sub(noise_stats->power[i], noise_stats->noise[i]);
327  snr = MFCC2FLOAT(noise_stats->power[i] - noise_stats->noise[i]);
328 #endif
329  if (snr > lrt)
330  lrt = snr;
331  }
332 
333  if (fe->remove_silence)
334  fe->vad_data->local_state = lrt > VAD_THRESHOLD;
335  else
336  fe->vad_data->local_state = 1;
337 
338  fe_low_envelope(noise_stats, signal, noise_stats->floor, num_filts);
339 
340  fe_temp_masking(noise_stats, signal, noise_stats->peak, num_filts);
341 
342  if (!fe->remove_noise) {
343  //no need for further calculations if noise cancellation disabled
344  ckd_free(signal);
345  return;
346  }
347 
348  for (i = 0; i < num_filts; i++) {
349  if (signal[i] < noise_stats->floor[i])
350  signal[i] = noise_stats->floor[i];
351  }
352 
353  gain = (powspec_t *) ckd_calloc(num_filts, sizeof(powspec_t));
354 #ifndef FIXED_POINT
355  for (i = 0; i < num_filts; i++) {
356  if (signal[i] < noise_stats->max_gain * noise_stats->power[i])
357  gain[i] = signal[i] / noise_stats->power[i];
358  else
359  gain[i] = noise_stats->max_gain;
360  if (gain[i] < noise_stats->inv_max_gain)
361  gain[i] = noise_stats->inv_max_gain;
362  }
363 #else
364  for (i = 0; i < num_filts; i++) {
365  gain[i] = signal[i] - noise_stats->power[i];
366  if (gain[i] > noise_stats->max_gain)
367  gain[i] = noise_stats->max_gain;
368  if (gain[i] < noise_stats->inv_max_gain)
369  gain[i] = noise_stats->inv_max_gain;
370  }
371 #endif
372 
373  /* Weight smoothing and time frequency normalization */
374  fe_weight_smooth(noise_stats, mfspec, gain, num_filts);
375 
376  ckd_free(gain);
377  ckd_free(signal);
378 }
