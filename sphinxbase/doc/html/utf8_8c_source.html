29 utf8_decode(uint32 *state, uint32 *codep, uint32 byte) {
30  uint32 type = utf8d[byte];
31 
32  *codep = (*state != UTF8_ACCEPT) ?
33  (byte & 0x3fu) | (*codep << 6) :
34  (0xff >> type) & (byte);
35 
36  *state = utf8d[256 + *state*16 + type];
37  return *state;
38 }
