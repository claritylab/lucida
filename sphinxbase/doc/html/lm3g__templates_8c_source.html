44 /* Locate a specific bigram within a bigram list */
45 #define BINARY_SEARCH_THRESH 16
46 static int32
47 find_bg(bigram_t * bg, int32 n, int32 w)
48 {
49  int32 i, b, e;
50 
51  /* Binary search until segment size < threshold */
52  b = 0;
53  e = n;
54  while (e - b > BINARY_SEARCH_THRESH) {
55  i = (b + e) >> 1;
56  if (bg[i].wid < w)
57  b = i + 1;
58  else if (bg[i].wid > w)
59  e = i;
60  else
61  return i;
62  }
63 
64  /* Linear search within narrowed segment */
65  for (i = b; (i < e) && (bg[i].wid != w); i++);
66  return ((i < e) ? i : -1);
67 }

69 static int32
70 lm3g_bg_score(NGRAM_MODEL_TYPE *model,
71  int32 lw1, int32 lw2, int32 *n_used)
72 {
73  int32 i, n, b, score;
74  bigram_t *bg;
75 
76  if (lw1 < 0 || model->base.n < 2) {
77  *n_used = 1;
78  return model->lm3g.unigrams[lw2].prob1.l;
79  }
80 
81  b = FIRST_BG(model, lw1);
82  n = FIRST_BG(model, lw1 + 1) - b;
83  bg = model->lm3g.bigrams + b;
84 
85  if ((i = find_bg(bg, n, lw2)) >= 0) {
86  /* Access mode = bigram */
87  *n_used = 2;
88  score = model->lm3g.prob2[bg[i].prob2].l;
89  }
90  else {
91  /* Access mode = unigram */
92  *n_used = 1;
93  score = model->lm3g.unigrams[lw1].bo_wt1.l + model->lm3g.unigrams[lw2].prob1.l;
94  }
95 
96  return (score);
97 }

99 static void
100 load_tginfo(NGRAM_MODEL_TYPE *model, int32 lw1, int32 lw2)
101 {
102  int32 i, n, b, t;
103  bigram_t *bg;
104  tginfo_t *tginfo;
105 
106  /* First allocate space for tg information for bg lw1,lw2 */
107  tginfo = (tginfo_t *) listelem_malloc(model->lm3g.le);
108  tginfo->w1 = lw1;
109  tginfo->tg = NULL;
110  tginfo->next = model->lm3g.tginfo[lw2];
111  model->lm3g.tginfo[lw2] = tginfo;
112 
113  /* Locate bigram lw1,lw2 */
114  b = model->lm3g.unigrams[lw1].bigrams;
115  n = model->lm3g.unigrams[lw1 + 1].bigrams - b;
116  bg = model->lm3g.bigrams + b;
117 
118  if ((n > 0) && ((i = find_bg(bg, n, lw2)) >= 0)) {
119  tginfo->bowt = model->lm3g.bo_wt2[bg[i].bo_wt2].l;
120 
121  /* Find t = Absolute first trigram index for bigram lw1,lw2 */
122  b += i; /* b = Absolute index of bigram lw1,lw2 on disk */
123  t = FIRST_TG(model, b);
124 
125  tginfo->tg = model->lm3g.trigrams + t;
126 
127  /* Find #tg for bigram w1,w2 */
128  tginfo->n_tg = FIRST_TG(model, b + 1) - t;
129  }
130  else { /* No bigram w1,w2 */
131  tginfo->bowt = 0;
132  tginfo->n_tg = 0;
133  }
134 }

158 static int32
159 lm3g_tg_score(NGRAM_MODEL_TYPE *model, int32 lw1,
160  int32 lw2, int32 lw3, int32 *n_used)
161 {
162  ngram_model_t *base = &model->base;
163  int32 i, n, score;
164  trigram_t *tg;
165  tginfo_t *tginfo, *prev_tginfo;
166 
167  if ((base->n < 3) || (lw1 < 0) || (lw2 < 0))
168  return (lm3g_bg_score(model, lw2, lw3, n_used));
169 
170  prev_tginfo = NULL;
171  for (tginfo = model->lm3g.tginfo[lw2]; tginfo; tginfo = tginfo->next) {
172  if (tginfo->w1 == lw1)
173  break;
174  prev_tginfo = tginfo;
175  }
176 
177  if (!tginfo) {
178  load_tginfo(model, lw1, lw2);
179  tginfo = model->lm3g.tginfo[lw2];
180  }
181  else if (prev_tginfo) {
182  prev_tginfo->next = tginfo->next;
183  tginfo->next = model->lm3g.tginfo[lw2];
184  model->lm3g.tginfo[lw2] = tginfo;
185  }
186 
187  tginfo->used = 1;
188 
189  /* Trigrams for w1,w2 now pointed to by tginfo */
190  n = tginfo->n_tg;
191  tg = tginfo->tg;
192  if ((i = find_tg(tg, n, lw3)) >= 0) {
193  /* Access mode = trigram */
194  *n_used = 3;
195  score = model->lm3g.prob3[tg[i].prob3].l;
196  }
197  else {
198  score = tginfo->bowt + lm3g_bg_score(model, lw2, lw3, n_used);
199  }
200 
201  return (score);
202 }

204 static int32
205 lm3g_template_score(ngram_model_t *base, int32 wid,
206  int32 *history, int32 n_hist,
207  int32 *n_used)
208 {
209  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)base;
210  switch (n_hist) {
211  case 0:
212  /* Access mode: unigram */
213  *n_used = 1;
214  return model->lm3g.unigrams[wid].prob1.l;
215  case 1:
216  return lm3g_bg_score(model, history[0], wid, n_used);
217  case 2:
218  default:
219  /* Anything greater than 2 is the same as a trigram for now. */
220  return lm3g_tg_score(model, history[1], history[0], wid, n_used);
221  }
222 }

224 static int32
225 lm3g_template_raw_score(ngram_model_t *base, int32 wid,
226  int32 *history, int32 n_hist,
227  int32 *n_used)
228 {
229  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)base;
230  int32 score;
231 
232  switch (n_hist) {
233  case 0:
234  /* Access mode: unigram */
235  *n_used = 1;
236  /* Undo insertion penalty. */
237  score = model->lm3g.unigrams[wid].prob1.l - base->log_wip;
238  /* Undo language weight. */
239  score = (int32)(score / base->lw);
240  /* Undo unigram interpolation */
241  if (strcmp(base->word_str[wid], "<s>") != 0) { /* FIXME: configurable start_sym */
242  /* This operation is numerically unstable, so try to avoid it
243  * as possible */
244  if (base->log_uniform + base->log_uniform_weight > logmath_get_zero(base->lmath)) {
245  score = logmath_log(base->lmath,
246  logmath_exp(base->lmath, score)
247  - logmath_exp(base->lmath,
248  base->log_uniform + base->log_uniform_weight));
249  }
250  }
251  return score;
252  case 1:
253  score = lm3g_bg_score(model, history[0], wid, n_used);
254  break;
255  case 2:
256  default:
257  /* Anything greater than 2 is the same as a trigram for now. */
258  score = lm3g_tg_score(model, history[1], history[0], wid, n_used);
259  break;
260  }
261  /* FIXME (maybe): This doesn't undo unigram weighting in backoff cases. */
262  return (int32)((score - base->log_wip) / base->lw);
263 }

287 static ngram_iter_t *
288 lm3g_template_iter(ngram_model_t *base, int32 wid,
289  int32 *history, int32 n_hist)
290 {
291  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)base;
292  lm3g_iter_t *itor = ckd_calloc(1, sizeof(*itor));
293 
294  ngram_iter_init((ngram_iter_t *)itor, base, n_hist, FALSE);
295 
296  if (n_hist == 0) {
297  /* Unigram is the easiest. */
298  itor->ug = model->lm3g.unigrams + wid;
299  return (ngram_iter_t *)itor;
300  }
301  else if (n_hist == 1) {
302  int32 i, n, b;
303  /* Find the bigram, as in bg_score above (duplicate code...) */
304  itor->ug = model->lm3g.unigrams + history[0];
305  b = FIRST_BG(model, history[0]);
306  n = FIRST_BG(model, history[0] + 1) - b;
307  itor->bg = model->lm3g.bigrams + b;
308  /* If no such bigram exists then fail. */
309  if ((i = find_bg(itor->bg, n, wid)) < 0) {
310  ngram_iter_free((ngram_iter_t *)itor);
311  return NULL;
312  }
313  itor->bg += i;
314  return (ngram_iter_t *)itor;
315  }
316  else if (n_hist == 2) {
317  int32 i, n;
318  tginfo_t *tginfo, *prev_tginfo;
319  /* Find the trigram, as in tg_score above (duplicate code...) */
320  itor->ug = model->lm3g.unigrams + history[1];
321  prev_tginfo = NULL;
322  for (tginfo = model->lm3g.tginfo[history[0]];
323  tginfo; tginfo = tginfo->next) {
324  if (tginfo->w1 == history[1])
325  break;
326  prev_tginfo = tginfo;
327  }
328 
329  if (!tginfo) {
330  load_tginfo(model, history[1], history[0]);
331  tginfo = model->lm3g.tginfo[history[0]];
332  }
333  else if (prev_tginfo) {
334  prev_tginfo->next = tginfo->next;
335  tginfo->next = model->lm3g.tginfo[history[0]];
336  model->lm3g.tginfo[history[0]] = tginfo;
337  }
338 
339  tginfo->used = 1;
340 
341  /* Trigrams for w1,w2 now pointed to by tginfo */
342  n = tginfo->n_tg;
343  itor->tg = tginfo->tg;
344  if ((i = find_tg(itor->tg, n, wid)) >= 0) {
345  itor->tg += i;
346  /* Now advance the bigram pointer accordingly. FIXME:
347  * Note that we actually already found the relevant bigram
348  * in load_tginfo. */
349  itor->bg = model->lm3g.bigrams;
350  while (FIRST_TG(model, (itor->bg - model->lm3g.bigrams + 1))
351  <= (itor->tg - model->lm3g.trigrams))
352  ++itor->bg;
353  return (ngram_iter_t *)itor;
354  }
355  else {
356  ngram_iter_free((ngram_iter_t *)itor);
357  return (ngram_iter_t *)NULL;
358  }
359  }
360  else {
361  /* Should not happen. */
362  assert(n_hist == 0); /* Guaranteed to fail. */
363  ngram_iter_free((ngram_iter_t *)itor);
364  return NULL;
365  }
366 }

368 static ngram_iter_t *
369 lm3g_template_mgrams(ngram_model_t *base, int m)
370 {
371  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)base;
372  lm3g_iter_t *itor = ckd_calloc(1, sizeof(*itor));
373  ngram_iter_init((ngram_iter_t *)itor, base, m, FALSE);
374 
375  itor->ug = model->lm3g.unigrams;
376  itor->bg = model->lm3g.bigrams;
377  itor->tg = model->lm3g.trigrams;
378 
379  /* Advance bigram pointer to match first trigram. */
380  if (m > 1 && base->n_counts[1] > 1) {
381  while (FIRST_TG(model, (itor->bg - model->lm3g.bigrams + 1))
382  <= (itor->tg - model->lm3g.trigrams))
383  ++itor->bg;
384  }
385 
386  /* Advance unigram pointer to match first bigram. */
387  if (m > 0 && base->n_counts[0] > 1) {
388  while (itor->ug[1].bigrams <= (itor->bg - model->lm3g.bigrams))
389  ++itor->ug;
390  }
391 
392  return (ngram_iter_t *)itor;
393 }

395 static ngram_iter_t *
396 lm3g_template_successors(ngram_iter_t *bitor)
397 {
398  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)bitor->model;
399  lm3g_iter_t *from = (lm3g_iter_t *)bitor;
400  lm3g_iter_t *itor = ckd_calloc(1, sizeof(*itor));
401 
402  itor->ug = from->ug;
403  switch (bitor->m) {
404  case 0:
405  /* Next itor bigrams is the same as this itor bigram or
406  itor bigrams is more than total count. This means no successors */
407  if (((itor->ug + 1) - model->lm3g.unigrams < bitor->model->n_counts[0] &&
408  itor->ug->bigrams == (itor->ug + 1)->bigrams) ||
409  itor->ug->bigrams == bitor->model->n_counts[1])
410  goto done;
411 
412  /* Start iterating from first bigram successor of from->ug. */
413  itor->bg = model->lm3g.bigrams + itor->ug->bigrams;
414  break;
415  case 1:
416  itor->bg = from->bg;
417 
418  /* This indicates no successors */
419  if (((itor->bg + 1) - model->lm3g.bigrams < bitor->model->n_counts[1] &&
420  FIRST_TG (model, itor->bg - model->lm3g.bigrams) ==
421  FIRST_TG (model, (itor->bg + 1) - model->lm3g.bigrams)) ||
422  FIRST_TG (model, itor->bg - model->lm3g.bigrams) == bitor->model->n_counts[2])
423  goto done;
424 
425  /* Start iterating from first trigram successor of from->bg. */
426  itor->tg = (model->lm3g.trigrams
427  + FIRST_TG(model, (itor->bg - model->lm3g.bigrams)));
428 #if 0
429  printf("%s %s => %d (%s)\n",
430  model->base.word_str[itor->ug - model->lm3g.unigrams],
431  model->base.word_str[itor->bg->wid],
432  FIRST_TG(model, (itor->bg - model->lm3g.bigrams)),
433  model->base.word_str[itor->tg->wid]);
434 #endif
435  break;
436  case 2:
437  default:
438  /* All invalid! */
439  goto done;
440  }
441 
442  ngram_iter_init((ngram_iter_t *)itor, bitor->model, bitor->m + 1, TRUE);
443  return (ngram_iter_t *)itor;
444  done:
445  ckd_free(itor);
446  return NULL;
447 }

449 static int32 const *
450 lm3g_template_iter_get(ngram_iter_t *base,
451  int32 *out_score, int32 *out_bowt)
452 {
453  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)base->model;
454  lm3g_iter_t *itor = (lm3g_iter_t *)base;
455 
456  base->wids[0] = itor->ug - model->lm3g.unigrams;
457  if (itor->bg) base->wids[1] = itor->bg->wid;
458  if (itor->tg) base->wids[2] = itor->tg->wid;
459 #if 0
460  printf("itor_get: %d %d %d\n", base->wids[0], base->wids[1], base->wids[2]);
461 #endif
462 
463  switch (base->m) {
464  case 0:
465  *out_score = itor->ug->prob1.l;
466  *out_bowt = itor->ug->bo_wt1.l;
467  break;
468  case 1:
469  *out_score = model->lm3g.prob2[itor->bg->prob2].l;
470  if (model->lm3g.bo_wt2)
471  *out_bowt = model->lm3g.bo_wt2[itor->bg->bo_wt2].l;
472  else
473  *out_bowt = 0;
474  break;
475  case 2:
476  *out_score = model->lm3g.prob3[itor->tg->prob3].l;
477  *out_bowt = 0;
478  break;
479  default: /* Should not happen. */
480  return NULL;
481  }
482  return base->wids;
483 }

485 static ngram_iter_t *
486 lm3g_template_iter_next(ngram_iter_t *base)
487 {
488  NGRAM_MODEL_TYPE *model = (NGRAM_MODEL_TYPE *)base->model;
489  lm3g_iter_t *itor = (lm3g_iter_t *)base;
490 
491  switch (base->m) {
492  case 0:
493  ++itor->ug;
494  /* Check for end condition. */
495  if (itor->ug - model->lm3g.unigrams >= base->model->n_counts[0])
496  goto done;
497  break;
498  case 1:
499  ++itor->bg;
500  /* Check for end condition. */
501  if (itor->bg - model->lm3g.bigrams >= base->model->n_counts[1])
502  goto done;
503  /* Advance unigram pointer if necessary in order to get one
504  * that points to this bigram. */
505  while (itor->bg - model->lm3g.bigrams >= itor->ug[1].bigrams) {
506  /* Stop if this is a successor iterator, since we don't
507  * want a new unigram. */
508  if (base->successor)
509  goto done;
510  ++itor->ug;
511  if (itor->ug == model->lm3g.unigrams + base->model->n_counts[0]) {
512  E_ERROR("Bigram %d has no valid unigram parent\n",
513  itor->bg - model->lm3g.bigrams);
514  goto done;
515  }
516  }
517  break;
518  case 2:
519  ++itor->tg;
520  /* Check for end condition. */
521  if (itor->tg - model->lm3g.trigrams >= base->model->n_counts[2])
522  goto done;
523  /* Advance bigram pointer if necessary. */
524  while (itor->tg - model->lm3g.trigrams >=
525  FIRST_TG(model, (itor->bg - model->lm3g.bigrams + 1))) {
526  if (base->successor)
527  goto done;
528  ++itor->bg;
529  if (itor->bg == model->lm3g.bigrams + base->model->n_counts[1]) {
530  E_ERROR("Trigram %d has no valid bigram parent\n",
531  itor->tg - model->lm3g.trigrams);
532 
533  goto done;
534  }
535  }
536  /* Advance unigram pointer if necessary. */
537  while (itor->bg - model->lm3g.bigrams >= itor->ug[1].bigrams) {
538  ++itor->ug;
539  if (itor->ug == model->lm3g.unigrams + base->model->n_counts[0]) {
540  E_ERROR("Trigram %d has no valid unigram parent\n",
541  itor->tg - model->lm3g.trigrams);
542  goto done;
543  }
544  }
545  break;
546  default: /* Should not happen. */
547  goto done;
548  }
549 
550  return (ngram_iter_t *)itor;
551 done:
552  ngram_iter_free(base);
553  return NULL;
554 }
