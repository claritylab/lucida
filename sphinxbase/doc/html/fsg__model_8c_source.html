89 static int32
90 nextline_str2words(FILE * fp, int32 * lineno,
91  char **lineptr, char ***wordptr)
92 {
93  for (;;) {
94  size_t len;
95  int32 n;
96 
97  ckd_free(*lineptr);
98  if ((*lineptr = fread_line(fp, &len)) == NULL)
99  return -1;
100 
101  (*lineno)++;
102 
103  if ((*lineptr)[0] == FSG_MODEL_COMMENT_CHAR)
104  continue; /* Skip comment lines */
105 
106  n = str2words(*lineptr, NULL, 0);
107  if (n == 0)
108  continue; /* Skip blank lines */
109 
110  /* Abuse of realloc(), but this doesn't have to be fast. */
111  if (*wordptr == NULL)
112  *wordptr = ckd_calloc(n, sizeof(**wordptr));
113  else
114  *wordptr = ckd_realloc(*wordptr, n * sizeof(**wordptr));
115  return str2words(*lineptr, *wordptr, n);
116  }
117 }

119 void
120 fsg_model_trans_add(fsg_model_t * fsg,
121  int32 from, int32 to, int32 logp, int32 wid)
122 {
123  fsg_link_t *link;
124  glist_t gl;
125  gnode_t *gn;
126 
127  if (fsg->trans[from].trans == NULL)
128  fsg->trans[from].trans = hash_table_new(5, HASH_CASE_YES);
129 
130  /* Check for duplicate link (i.e., link already exists with label=wid) */
131  for (gn = gl = fsg_model_trans(fsg, from, to); gn; gn = gnode_next(gn)) {
132  link = (fsg_link_t *) gnode_ptr(gn);
133  if (link->wid == wid) {
134  if (link->logs2prob < logp)
135  link->logs2prob = logp;
136  return;
137  }
138  }
139 
140  /* Create transition object */
141  link = listelem_malloc(fsg->link_alloc);
142  link->from_state = from;
143  link->to_state = to;
144  link->logs2prob = logp;
145  link->wid = wid;
146 
147  /* Add it to the list of transitions and update the hash table */
148  gl = glist_add_ptr(gl, (void *) link);
149  hash_table_replace_bkey(fsg->trans[from].trans,
150  (char const *) &link->to_state,
151  sizeof(link->to_state), gl);
152 }

154 int32
155 fsg_model_tag_trans_add(fsg_model_t * fsg, int32 from, int32 to,
156  int32 logp, int32 wid)
157 {
158  fsg_link_t *link, *link2;
159 
160  /* Check for transition probability */
161  if (logp > 0) {
162  E_FATAL("Null transition prob must be <= 1.0 (state %d -> %d)\n",
163  from, to);
164  }
165 
166  /* Self-loop null transitions (with prob <= 1.0) are redundant */
167  if (from == to)
168  return -1;
169 
170  if (fsg->trans[from].null_trans == NULL)
171  fsg->trans[from].null_trans = hash_table_new(5, HASH_CASE_YES);
172 
173  /* Check for a duplicate link; if found, keep the higher prob */
174  link = fsg_model_null_trans(fsg, from, to);
175  if (link) {
176  if (link->logs2prob < logp) {
177  link->logs2prob = logp;
178  return 0;
179  }
180  else
181  return -1;
182  }
183 
184  /* Create null transition object */
185  link = listelem_malloc(fsg->link_alloc);
186  link->from_state = from;
187  link->to_state = to;
188  link->logs2prob = logp;
189  link->wid = -1;
190 
191  link2 = (fsg_link_t *)
192  hash_table_enter_bkey(fsg->trans[from].null_trans,
193  (char const *) &link->to_state,
194  sizeof(link->to_state), link);
195  assert(link == link2);
196 
197  return 1;
198 }

207 glist_t
208 fsg_model_null_trans_closure(fsg_model_t * fsg, glist_t nulls)
209 {
210  gnode_t *gn1;
211  int updated;
212  fsg_link_t *tl1, *tl2;
213  int32 k, n;
214 
215  E_INFO("Computing transitive closure for null transitions\n");
216 
217  /* If our caller didn't give us a list of null-transitions,
218  make such a list. Just loop through all the FSG states,
219  and all the null-transitions in that state (which are kept in
220  their own hash table). */
221  if (nulls == NULL) {
222  int i;
223  for (i = 0; i < fsg->n_state; ++i) {
224  hash_iter_t *itor;
225  hash_table_t *null_trans = fsg->trans[i].null_trans;
226  if (null_trans == NULL)
227  continue;
228  for (itor = hash_table_iter(null_trans);
229  itor != NULL;
230  itor = hash_table_iter_next(itor)) {
231  nulls = glist_add_ptr(nulls, hash_entry_val(itor->ent));
232  }
233  }
234  }
235 
236  /*
237  * Probably not the most efficient closure implementation, in general, but
238  * probably reasonably efficient for a sparse null transition matrix.
239  */
240  n = 0;
241  do {
242  updated = FALSE;
243 
244  for (gn1 = nulls; gn1; gn1 = gnode_next(gn1)) {
245  hash_iter_t *itor;
246 
247  tl1 = (fsg_link_t *) gnode_ptr(gn1);
248  assert(tl1->wid < 0);
249 
250  if (fsg->trans[tl1->to_state].null_trans == NULL)
251  continue;
252 
253  for (itor = hash_table_iter(fsg->trans[tl1->to_state].null_trans);
254  itor; itor = hash_table_iter_next(itor)) {
255 
256  tl2 = (fsg_link_t *) hash_entry_val(itor->ent);
257 
258  k = fsg_model_null_trans_add(fsg,
259  tl1->from_state,
260  tl2->to_state,
261  tl1->logs2prob +
262  tl2->logs2prob);
263  if (k >= 0) {
264  updated = TRUE;
265  if (k > 0) {
266  nulls = glist_add_ptr(nulls, (void *)
267  fsg_model_null_trans
268  (fsg, tl1->from_state,
269  tl2->to_state));
270  n++;
271  }
272  }
273  }
274  }
275  } while (updated);
276 
277  E_INFO("%d null transitions added\n", n);
278 
279  return nulls;
280 }

283 fsg_model_trans(fsg_model_t * fsg, int32 i, int32 j)
284 {
285  void *val;
286 
287  if (fsg->trans[i].trans == NULL)
288  return NULL;
289  if (hash_table_lookup_bkey(fsg->trans[i].trans, (char const *) &j,

291  return NULL;
292  return (glist_t) val;

308 fsg_arciter_t *
309 fsg_model_arcs(fsg_model_t * fsg, int32 i)
310 {
311  fsg_arciter_t *itor;
312 
313  if (fsg->trans[i].trans == NULL && fsg->trans[i].null_trans == NULL)
314  return NULL;
315  itor = ckd_calloc(1, sizeof(*itor));
316  if (fsg->trans[i].null_trans)
317  itor->null_itor = hash_table_iter(fsg->trans[i].null_trans);
318  if (fsg->trans[i].trans)
319  itor->itor = hash_table_iter(fsg->trans[i].trans);
320  if (itor->itor != NULL)
321  itor->gn = hash_entry_val(itor->itor->ent);
322  return itor;
323 }

325 fsg_link_t *
326 fsg_arciter_get(fsg_arciter_t * itor)
327 {
328  /* Iterate over non-null arcs first. */
329  if (itor->gn)
330  return (fsg_link_t *) gnode_ptr(itor->gn);
331  else if (itor->null_itor)
332  return (fsg_link_t *) hash_entry_val(itor->null_itor->ent);
333  else
334  return NULL;
335 }
336 
337 fsg_arciter_t *
338 fsg_arciter_next(fsg_arciter_t * itor)
339 {
340  /* Iterate over non-null arcs first. */
341  if (itor->gn) {
342  itor->gn = gnode_next(itor->gn);
343  /* Move to the next destination arc. */
344  if (itor->gn == NULL) {
345  itor->itor = hash_table_iter_next(itor->itor);
346  if (itor->itor != NULL)
347  itor->gn = hash_entry_val(itor->itor->ent);
348  else if (itor->null_itor == NULL)
349  goto stop_iteration;
350  }
351  }
352  else {
353  if (itor->null_itor == NULL)
354  goto stop_iteration;
355  itor->null_itor = hash_table_iter_next(itor->null_itor);
356  if (itor->null_itor == NULL)
357  goto stop_iteration;
358  }
359  return itor;
360  stop_iteration:
361  fsg_arciter_free(itor);
362  return NULL;
363 
364 }

376 int
377 fsg_model_word_id(fsg_model_t * fsg, char const *word)
378 {
379  int wid;
380 
381  /* Search for an existing word matching this. */
382  for (wid = 0; wid < fsg->n_word; ++wid) {
383  if (0 == strcmp(fsg->vocab[wid], word))
384  break;
385  }
386  /* If not found, add this to the vocab. */
387  if (wid == fsg->n_word)
388  return -1;
389  return wid;
390 }

392 int
393 fsg_model_word_add(fsg_model_t * fsg, char const *word)
394 {
395  int wid, old_size;
396 
397  /* Search for an existing word matching this. */
398  wid = fsg_model_word_id(fsg, word);
399  /* If not found, add this to the vocab. */
400  if (wid == -1) {
401  wid = fsg->n_word;
402  if (fsg->n_word == fsg->n_word_alloc) {
403  old_size = fsg->n_word_alloc;
404  fsg->n_word_alloc += 10;
405  fsg->vocab = ckd_realloc(fsg->vocab,
406  fsg->n_word_alloc *
407  sizeof(*fsg->vocab));
408  if (fsg->silwords)
409  fsg->silwords =
410  bitvec_realloc(fsg->silwords, old_size, fsg->n_word_alloc);
411  if (fsg->altwords)
412  fsg->altwords =
413  bitvec_realloc(fsg->altwords, old_size, fsg->n_word_alloc);
414  }
415  ++fsg->n_word;
416  fsg->vocab[wid] = ckd_salloc(word);
417  }
418  return wid;
419 }

421 int
422 fsg_model_add_silence(fsg_model_t * fsg, char const *silword,
423  int state, float32 silprob)
424 {
425  int32 logsilp;
426  int n_trans, silwid, src;
427 
428  E_INFO("Adding silence transitions for %s to FSG\n", silword);
429 
430  silwid = fsg_model_word_add(fsg, silword);
431  logsilp = (int32) (logmath_log(fsg->lmath, silprob) * fsg->lw);
432  if (fsg->silwords == NULL)
433  fsg->silwords = bitvec_alloc(fsg->n_word_alloc);
434  bitvec_set(fsg->silwords, silwid);
435 
436  n_trans = 0;
437  if (state == -1) {
438  for (src = 0; src < fsg->n_state; src++) {
439  fsg_model_trans_add(fsg, src, src, logsilp, silwid);
440  ++n_trans;
441  }
442  }
443  else {
444  fsg_model_trans_add(fsg, state, state, logsilp, silwid);
445  ++n_trans;
446  }
447 
448  E_INFO("Added %d silence word transitions\n", n_trans);
449  return n_trans;
450 }

452 int
453 fsg_model_add_alt(fsg_model_t * fsg, char const *baseword,
454  char const *altword)
455 {
456  int i, basewid, altwid;
457  int ntrans;
458 
459  /* FIXME: This will get slow, eventually... */
460  for (basewid = 0; basewid < fsg->n_word; ++basewid)
461  if (0 == strcmp(fsg->vocab[basewid], baseword))
462  break;
463  if (basewid == fsg->n_word) {
464  E_ERROR("Base word %s not present in FSG vocabulary!\n", baseword);
465  return -1;
466  }
467  altwid = fsg_model_word_add(fsg, altword);
468  if (fsg->altwords == NULL)
469  fsg->altwords = bitvec_alloc(fsg->n_word_alloc);
470  bitvec_set(fsg->altwords, altwid);
471  if (fsg_model_is_filler(fsg, basewid)) {
472  if (fsg->silwords == NULL)
473  fsg->silwords = bitvec_alloc(fsg->n_word_alloc);
474  bitvec_set(fsg->silwords, altwid);
475  }
476 
477  E_DEBUG(2, ("Adding alternate word transitions (%s,%s) to FSG\n",
478  baseword, altword));
479 
480  /* Look for all transitions involving baseword and duplicate them. */
481  /* FIXME: This will also get slow, eventually... */
482  ntrans = 0;
483  for (i = 0; i < fsg->n_state; ++i) {
484  hash_iter_t *itor;
485  if (fsg->trans[i].trans == NULL)
486  continue;
487  for (itor = hash_table_iter(fsg->trans[i].trans); itor;
488  itor = hash_table_iter_next(itor)) {
489  glist_t trans;
490  gnode_t *gn;
491 
492  trans = hash_entry_val(itor->ent);
493  for (gn = trans; gn; gn = gnode_next(gn)) {
494  fsg_link_t *fl = gnode_ptr(gn);
495  if (fl->wid == basewid) {
496  fsg_link_t *link;
497 
498  /* Create transition object */
499  link = listelem_malloc(fsg->link_alloc);
500  link->from_state = fl->from_state;
501  link->to_state = fl->to_state;
502  link->logs2prob = fl->logs2prob; /* FIXME!!!??? */
503  link->wid = altwid;
504 
505  trans = glist_add_ptr(trans, (void *) link);
506  ++ntrans;
507  }
508  }
509  hash_entry_val(itor->ent) = trans;
510  }
511  }
512 
513  E_DEBUG(2, ("Added %d alternate word transitions\n", ntrans));
514  return ntrans;
515 }

538 fsg_model_t *
539 fsg_model_read(FILE * fp, logmath_t * lmath, float32 lw)
540 {
541  fsg_model_t *fsg;
542  hash_table_t *vocab;
543  hash_iter_t *itor;
544  int32 lastwid;
545  char **wordptr;
546  char *lineptr;
547  char *fsgname;
548  int32 lineno;
549  int32 n, i, j;
550  int n_state, n_trans, n_null_trans;
551  glist_t nulls;
552  float32 p;
553 
554  lineno = 0;
555  vocab = hash_table_new(32, FALSE);
556  wordptr = NULL;
557  lineptr = NULL;
558  nulls = NULL;
559  fsgname = NULL;
560  fsg = NULL;
561 
562  /* Scan upto FSG_BEGIN header */
563  for (;;) {
564  n = nextline_str2words(fp, &lineno, &lineptr, &wordptr);
565  if (n < 0) {
566  E_ERROR("%s declaration missing\n", FSG_MODEL_BEGIN_DECL);
567  goto parse_error;
568  }
569 
570  if ((strcmp(wordptr[0], FSG_MODEL_BEGIN_DECL) == 0)) {
571  if (n > 2) {
572  E_ERROR("Line[%d]: malformed FSG_BEGIN declaration\n",
573  lineno);
574  goto parse_error;
575  }
576  break;
577  }
578  }
579  /* Save FSG name, or it will get clobbered below :(.
580  * If name is missing, try the default.
581  */
582  if (n == 2) {
583  fsgname = ckd_salloc(wordptr[1]);
584  }
585  else {
586  E_WARN("FSG name is missing\n");
587  fsgname = ckd_salloc("unknown");
588  }
589 
590  /* Read #states */
591  n = nextline_str2words(fp, &lineno, &lineptr, &wordptr);
592  if ((n != 2)
593  || ((strcmp(wordptr[0], FSG_MODEL_N_DECL) != 0)
594  && (strcmp(wordptr[0], FSG_MODEL_NUM_STATES_DECL) != 0))
595  || (sscanf(wordptr[1], "%d", &n_state) != 1)
596  || (n_state <= 0)) {
597  E_ERROR
598  ("Line[%d]: #states declaration line missing or malformed\n",
599  lineno);
600  goto parse_error;
601  }
602 
603  /* Now create the FSG. */
604  fsg = fsg_model_init(fsgname, lmath, lw, n_state);
605  ckd_free(fsgname);
606  fsgname = NULL;
607 
608  /* Read start state */
609  n = nextline_str2words(fp, &lineno, &lineptr, &wordptr);
610  if ((n != 2)
611  || ((strcmp(wordptr[0], FSG_MODEL_S_DECL) != 0)
612  && (strcmp(wordptr[0], FSG_MODEL_START_STATE_DECL) != 0))
613  || (sscanf(wordptr[1], "%d", &(fsg->start_state)) != 1)
614  || (fsg->start_state < 0)
615  || (fsg->start_state >= fsg->n_state)) {
616  E_ERROR
617  ("Line[%d]: start state declaration line missing or malformed\n",
618  lineno);
619  goto parse_error;
620  }
621 
622  /* Read final state */
623  n = nextline_str2words(fp, &lineno, &lineptr, &wordptr);
624  if ((n != 2)
625  || ((strcmp(wordptr[0], FSG_MODEL_F_DECL) != 0)
626  && (strcmp(wordptr[0], FSG_MODEL_FINAL_STATE_DECL) != 0))
627  || (sscanf(wordptr[1], "%d", &(fsg->final_state)) != 1)
628  || (fsg->final_state < 0)
629  || (fsg->final_state >= fsg->n_state)) {
630  E_ERROR
631  ("Line[%d]: final state declaration line missing or malformed\n",
632  lineno);
633  goto parse_error;
634  }
635 
636  /* Read transitions */
637  lastwid = 0;
638  n_trans = n_null_trans = 0;
639  for (;;) {
640  int32 wid, tprob;
641 
642  n = nextline_str2words(fp, &lineno, &lineptr, &wordptr);
643  if (n <= 0) {
644  E_ERROR("Line[%d]: transition or FSG_END statement expected\n",
645  lineno);
646  goto parse_error;
647  }
648 
649  if ((strcmp(wordptr[0], FSG_MODEL_END_DECL) == 0)) {
650  break;
651  }
652 
653  if ((strcmp(wordptr[0], FSG_MODEL_T_DECL) == 0)
654  || (strcmp(wordptr[0], FSG_MODEL_TRANSITION_DECL) == 0)) {
655 
656 
657  if (((n != 4) && (n != 5))
658  || (sscanf(wordptr[1], "%d", &i) != 1)
659  || (sscanf(wordptr[2], "%d", &j) != 1)
660  || (i < 0) || (i >= fsg->n_state)
661  || (j < 0) || (j >= fsg->n_state)) {
662  E_ERROR
663  ("Line[%d]: transition spec malformed; Expecting: from-state to-state trans-prob [word]\n",
664  lineno);
665  goto parse_error;
666  }
667 
668  p = atof_c(wordptr[3]);
669  if ((p <= 0.0) || (p > 1.0)) {
670  E_ERROR
671  ("Line[%d]: transition spec malformed; Expecting float as transition probability\n",
672  lineno);
673  goto parse_error;
674  }
675  }
676  else {
677  E_ERROR("Line[%d]: transition or FSG_END statement expected\n",
678  lineno);
679  goto parse_error;
680  }
681 
682  tprob = (int32) (logmath_log(lmath, p) * fsg->lw);
683  /* Add word to "dictionary". */
684  if (n > 4) {
685  if (hash_table_lookup_int32(vocab, wordptr[4], &wid) < 0) {
686  (void) hash_table_enter_int32(vocab,
687  ckd_salloc(wordptr[4]),
688  lastwid);
689  wid = lastwid;
690  ++lastwid;
691  }
692  fsg_model_trans_add(fsg, i, j, tprob, wid);
693  ++n_trans;
694  }
695  else {
696  if (fsg_model_null_trans_add(fsg, i, j, tprob) == 1) {
697  ++n_null_trans;
698  nulls =
699  glist_add_ptr(nulls, fsg_model_null_trans(fsg, i, j));
700  }
701  }
702  }
703 
704  E_INFO("FSG: %d states, %d unique words, %d transitions (%d null)\n",
705  fsg->n_state, hash_table_inuse(vocab), n_trans, n_null_trans);
706 
707 
708  /* Now create a string table from the "dictionary" */
709  fsg->n_word = hash_table_inuse(vocab);
710  fsg->n_word_alloc = fsg->n_word + 10; /* Pad it a bit. */
711  fsg->vocab = ckd_calloc(fsg->n_word_alloc, sizeof(*fsg->vocab));
712  for (itor = hash_table_iter(vocab); itor;
713  itor = hash_table_iter_next(itor)) {
714  char const *word = hash_entry_key(itor->ent);
715  int32 wid = (int32) (long) hash_entry_val(itor->ent);
716  fsg->vocab[wid] = (char *) word;
717  }
718  hash_table_free(vocab);
719 
720  /* Do transitive closure on null transitions */
721  nulls = fsg_model_null_trans_closure(fsg, nulls);
722  glist_free(nulls);
723 
724  ckd_free(lineptr);
725  ckd_free(wordptr);
726 
727  return fsg;
728 
729  parse_error:
730  for (itor = hash_table_iter(vocab); itor;
731  itor = hash_table_iter_next(itor))
732  ckd_free((char *) hash_entry_key(itor->ent));
733  glist_free(nulls);
734  hash_table_free(vocab);
735  ckd_free(fsgname);
736  ckd_free(lineptr);
737  ckd_free(wordptr);
738  fsg_model_free(fsg);
739  return NULL;
740 }
