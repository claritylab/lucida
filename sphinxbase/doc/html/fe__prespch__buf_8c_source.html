51 struct prespch_buf_s {
52  /* saved mfcc frames */
53  mfcc_t **cep_buf;
54  /* saved pcm audio */
55  int16 *pcm_buf;
56  /* flag for pcm buffer initialization */
57  uint8 pcm_init;
58  /* write pointer for cep buffer */
59  int16 cep_write_ptr;
60  /* read pointer for cep buffer */
61  int16 cep_read_ptr;
62  /* write pointer for pcm buffer */
63  int16 pcm_write_ptr;
64  /* frames amount in cep buffer */
65  int16 num_frames_cep;
66  /* frames amount in pcm buffer */
67  int16 num_frames_pcm;
68  /* filters amount */
69  int16 num_cepstra;
70  /* amount of fresh samples in frame */
71  int16 num_samples;
72 };

74 prespch_buf_t *
75 fe_init_prespch(int num_frames, int num_cepstra, int num_samples)
76 {
77  prespch_buf_t *prespch_buf;
78 
79  prespch_buf = (prespch_buf_t *) ckd_calloc(1, sizeof(prespch_buf_t));
80 
81  prespch_buf->num_cepstra = num_cepstra;
82  prespch_buf->num_frames_cep = num_frames;
83  prespch_buf->num_samples = num_samples;
84  prespch_buf->num_frames_pcm = 0;
85  prespch_buf->cep_write_ptr = 0;
86  prespch_buf->cep_read_ptr = 0;
87  prespch_buf->pcm_write_ptr = 0;
88  prespch_buf->pcm_init = 0;
89 
90  prespch_buf->cep_buf = (mfcc_t **)
91  ckd_calloc_2d(num_frames, num_cepstra,
92  sizeof(**prespch_buf->cep_buf));
93 
94  return prespch_buf;
95 }

97 void
98 fe_reinit_prespch_pcm(prespch_buf_t* prespch_buf, int num_frames_pcm)
99 {
100  num_frames_pcm += prespch_buf->num_frames_cep;
101  if (num_frames_pcm > prespch_buf->num_frames_pcm) {
102  if (prespch_buf->pcm_init && prespch_buf->pcm_buf)
103  ckd_free(prespch_buf->pcm_buf);
104  prespch_buf->num_frames_pcm = num_frames_pcm;
105  prespch_buf->pcm_buf = (int16 *) ckd_calloc(prespch_buf->num_frames_pcm * prespch_buf->num_samples, sizeof(int16));
106  prespch_buf->pcm_init = 1;
107  }
108 }

110 int
111 fe_prespch_read_cep(prespch_buf_t * prespch_buf, mfcc_t * fea)
112 {
113  if (prespch_buf->cep_read_ptr >= prespch_buf->num_frames_cep)
114  return 0;
115  if (prespch_buf->cep_read_ptr >= prespch_buf->cep_write_ptr)
116  return 0;
117  memcpy(fea, prespch_buf->cep_buf[prespch_buf->cep_read_ptr],
118  sizeof(mfcc_t) * prespch_buf->num_cepstra);
119  prespch_buf->cep_read_ptr++;
120  return 1;
121 }

123 void
124 fe_prespch_write_cep(prespch_buf_t * prespch_buf, mfcc_t * fea)
125 {
126  assert(prespch_buf->cep_write_ptr < prespch_buf->num_frames_cep);
127  memcpy(prespch_buf->cep_buf[prespch_buf->cep_write_ptr], fea,
128  sizeof(mfcc_t) * prespch_buf->num_cepstra);
129  prespch_buf->cep_write_ptr++;
130 }

132 void
133 fe_prespch_read_pcm(prespch_buf_t * prespch_buf, int16 ** samples,
134  int32 * samples_num)
135 {
136  if (!prespch_buf->pcm_init) {
137  /* pcm prespch buffer isn't initialized yet */
138  samples = NULL;
139  *samples_num = 0;
140  return;
141  }
142  *samples = prespch_buf->pcm_buf;
143  *samples_num = prespch_buf->pcm_write_ptr * prespch_buf->num_samples;
144  prespch_buf->pcm_write_ptr = 0;
145 }

147 void
148 fe_prespch_write_pcm(prespch_buf_t * prespch_buf, int16 * samples)
149 {
150  int32 sample_ptr;
151 
152  assert(prespch_buf->pcm_write_ptr < prespch_buf->num_frames_pcm);
153  sample_ptr = prespch_buf->pcm_write_ptr * prespch_buf->num_samples;
154  memcpy(&prespch_buf->pcm_buf[sample_ptr], samples,
155  prespch_buf->num_samples * sizeof(int16));
156  prespch_buf->pcm_write_ptr++;
157 }
