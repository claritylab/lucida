75 static void
76 eval_topn(s2_semi_mgau_t *s, int32 feat, mfcc_t *z)
77 {
78  int i, ceplen;
79  vqFeature_t *topn;
80 
81  topn = s->f[feat];
82  ceplen = s->veclen[feat];
83 
84  for (i = 0; i < s->max_topn; i++) {
85  mfcc_t *mean, diff, sqdiff, compl; /* diff, diff^2, component likelihood */
86  vqFeature_t vtmp;
87  mfcc_t *var, d;
88  mfcc_t *obs;
89  int32 cw, j;
90 
91  cw = topn[i].codeword;
92  mean = s->means[feat][0] + cw * ceplen;
93  var = s->vars[feat][0] + cw * ceplen;
94  d = s->dets[feat][cw];
95  obs = z;
96  for (j = 0; j < ceplen; j++) {
97  diff = *obs++ - *mean++;
98  sqdiff = MFCCMUL(diff, diff);
99  compl = MFCCMUL(sqdiff, *var);
100  d = GMMSUB(d, compl);
101  ++var;
102  }
103  topn[i].score = (int32)d;
104  if (i == 0)
105  continue;
106  vtmp = topn[i];
107  for (j = i - 1; j >= 0 && (int32)d > topn[j].score; j--) {
108  topn[j + 1] = topn[j];
109  }
110  topn[j + 1] = vtmp;
111  }
112 }

114 static void
115 eval_cb(s2_semi_mgau_t *s, int32 feat, mfcc_t *z)
116 {
117  vqFeature_t *worst, *best, *topn;
118  mfcc_t *mean;
119  mfcc_t *var, *det, *detP, *detE;
120  int32 i, ceplen;
121 
122  best = topn = s->f[feat];
123  worst = topn + (s->max_topn - 1);
124  mean = s->means[feat][0];
125  var = s->vars[feat][0];
126  det = s->dets[feat];
127  detE = det + s->n_density;
128  ceplen = s->veclen[feat];
129 
130  for (detP = det; detP < detE; ++detP) {
131  mfcc_t diff, sqdiff, compl; /* diff, diff^2, component likelihood */
132  mfcc_t d;
133  mfcc_t *obs;
134  vqFeature_t *cur;
135  int32 cw, j;
136 
137  d = *detP;
138  obs = z;
139  cw = detP - det;
140  for (j = 0; (j < ceplen) && (d >= worst->score); ++j) {
141  diff = *obs++ - *mean++;
142  sqdiff = MFCCMUL(diff, diff);
143  compl = MFCCMUL(sqdiff, *var);
144  d = GMMSUB(d, compl);
145  ++var;
146  }
147  if (j < ceplen) {
148  /* terminated early, so not in topn */
149  mean += (ceplen - j);
150  var += (ceplen - j);
151  continue;
152  }
153  if ((int32)d < worst->score)
154  continue;
155  for (i = 0; i < s->max_topn; i++) {
156  /* already there, so don't need to insert */
157  if (topn[i].codeword == cw)
158  break;
159  }
160  if (i < s->max_topn)
161  continue; /* already there. Don't insert */
162  /* remaining code inserts codeword and dist in correct spot */
163  for (cur = worst - 1; cur >= best && (int32)d >= cur->score; --cur)
164  memcpy(cur + 1, cur, sizeof(vqFeature_t));
165  ++cur;
166  cur->codeword = cw;
167  cur->score = (int32)d;
168  }
169 }

171 static void
172 mgau_dist(s2_semi_mgau_t * s, int32 frame, int32 feat, mfcc_t * z)
173 {
174  eval_topn(s, feat, z);
175 
176  /* If this frame is skipped, do nothing else. */
177  if (frame % s->ds_ratio)
178  return;
179 
180  /* Evaluate the rest of the codebook (or subset thereof). */
181  eval_cb(s, feat, z);
182 }

184 static int
185 mgau_norm(s2_semi_mgau_t *s, int feat)
186 {
187  int32 norm;
188  int j;
189 
190  /* Compute quantized normalizing constant. */
191  norm = s->f[feat][0].score >> SENSCR_SHIFT;
192 
193  /* Normalize the scores, negate them, and clamp their dynamic range. */
194  for (j = 0; j < s->max_topn; ++j) {
195  s->f[feat][j].score = -((s->f[feat][j].score >> SENSCR_SHIFT) - norm);
196  if (s->f[feat][j].score > MAX_NEG_ASCR)
197  s->f[feat][j].score = MAX_NEG_ASCR;
198  if (s->topn_beam[feat] && s->f[feat][j].score > s->topn_beam[feat])
199  break;
200  }
201  return j;
202 }

883 static int32
884 read_sendump(s2_semi_mgau_t *s, bin_mdef_t *mdef, char const *file)
885 {
886  FILE *fp;
887  char line[1000];
888  int32 i, n, r, c;
889  int32 do_swap, do_mmap;
890  size_t offset;
891  int n_clust = 0;
892  int n_feat = s->n_feat;
893  int n_density = s->n_density;
894  int n_sen = bin_mdef_n_sen(mdef);
895  int n_bits = 8;
896 
897  s->n_sen = n_sen; /* FIXME: Should have been done earlier */
898  do_mmap = cmd_ln_boolean_r(s->config, "-mmap");
899 
900  if ((fp = fopen(file, "rb")) == NULL)
901  return -1;
902 
903  E_INFO("Loading senones from dump file %s\n", file);
904  /* Read title size, title */
905  if (fread(&n, sizeof(int32), 1, fp) != 1) {
906  E_ERROR_SYSTEM("Failed to read title size from %s", file);
907  goto error_out;
908  }
909  /* This is extremely bogus */
910  do_swap = 0;
911  if (n < 1 || n > 999) {
912  SWAP_INT32(&n);
913  if (n < 1 || n > 999) {
914  E_ERROR("Title length %x in dump file %s out of range\n", n, file);
915  goto error_out;
916  }
917  do_swap = 1;
918  }
919  if (fread(line, sizeof(char), n, fp) != n) {
920  E_ERROR_SYSTEM("Cannot read title");
921  goto error_out;
922  }
923  if (line[n - 1] != '\0') {
924  E_ERROR("Bad title in dump file\n");
925  goto error_out;
926  }
927  E_INFO("%s\n", line);
928 
929  /* Read header size, header */
930  if (fread(&n, sizeof(n), 1, fp) != 1) {
931  E_ERROR_SYSTEM("Failed to read header size from %s", file);
932  goto error_out;
933  }
934  if (do_swap) SWAP_INT32(&n);
935  if (fread(line, sizeof(char), n, fp) != n) {
936  E_ERROR_SYSTEM("Cannot read header");
937  goto error_out;
938  }
939  if (line[n - 1] != '\0') {
940  E_ERROR("Bad header in dump file\n");
941  goto error_out;
942  }
943 
944  /* Read other header strings until string length = 0 */
945  for (;;) {
946  if (fread(&n, sizeof(n), 1, fp) != 1) {
947  E_ERROR_SYSTEM("Failed to read header string size from %s", file);
948  goto error_out;
949  }
950  if (do_swap) SWAP_INT32(&n);
951  if (n == 0)
952  break;
953  if (fread(line, sizeof(char), n, fp) != n) {
954  E_ERROR_SYSTEM("Cannot read header");
955  goto error_out;
956  }
957  /* Look for a cluster count, if present */
958  if (!strncmp(line, "feature_count ", strlen("feature_count "))) {
959  n_feat = atoi(line + strlen("feature_count "));
960  }
961  if (!strncmp(line, "mixture_count ", strlen("mixture_count "))) {
962  n_density = atoi(line + strlen("mixture_count "));
963  }
964  if (!strncmp(line, "model_count ", strlen("model_count "))) {
965  n_sen = atoi(line + strlen("model_count "));
966  }
967  if (!strncmp(line, "cluster_count ", strlen("cluster_count "))) {
968  n_clust = atoi(line + strlen("cluster_count "));
969  }
970  if (!strncmp(line, "cluster_bits ", strlen("cluster_bits "))) {
971  n_bits = atoi(line + strlen("cluster_bits "));
972  }
973  }
974 
975  /* Defaults for #rows, #columns in mixw array. */
976  c = n_sen;
977  r = n_density;
978  if (n_clust == 0) {
979  /* Older mixw files have them here, and they might be padded. */
980  if (fread(&r, sizeof(r), 1, fp) != 1) {
981  E_ERROR_SYSTEM("Cannot read #rows");
982  goto error_out;
983  }
984  if (do_swap) SWAP_INT32(&r);
985  if (fread(&c, sizeof(c), 1, fp) != 1) {
986  E_ERROR_SYSTEM("Cannot read #columns");
987  goto error_out;
988  }
989  if (do_swap) SWAP_INT32(&c);
990  E_INFO("Rows: %d, Columns: %d\n", r, c);
991  }
992 
993  if (n_feat != s->n_feat) {
994  E_ERROR("Number of feature streams mismatch: %d != %d\n",
995  n_feat, s->n_feat);
996  goto error_out;
997  }
998  if (n_density != s->n_density) {
999  E_ERROR("Number of densities mismatch: %d != %d\n",
1000  n_density, s->n_density);
1001  goto error_out;
1002  }
1003  if (n_sen != s->n_sen) {
1004  E_ERROR("Number of senones mismatch: %d != %d\n",
1005  n_sen, s->n_sen);
1006  goto error_out;
1007  }
1008 
1009  if (!((n_clust == 0) || (n_clust == 15) || (n_clust == 16))) {
1010  E_ERROR("Cluster count must be 0, 15, or 16\n");
1011  goto error_out;
1012  }
1013  if (n_clust == 15)
1014  ++n_clust;
1015 
1016  if (!((n_bits == 8) || (n_bits == 4))) {
1017  E_ERROR("Cluster count must be 4 or 8\n");
1018  goto error_out;
1019  }
1020 
1021  if (do_mmap) {
1022  E_INFO("Using memory-mapped I/O for senones\n");
1023  }
1024  offset = ftell(fp);
1025 
1026  /* Allocate memory for pdfs (or memory map them) */
1027  if (do_mmap) {
1028  s->sendump_mmap = mmio_file_read(file);
1029  /* Get cluster codebook if any. */
1030  if (n_clust) {
1031  s->mixw_cb = ((uint8 *) mmio_file_ptr(s->sendump_mmap)) + offset;
1032  offset += n_clust;
1033  }
1034  }
1035  else {
1036  /* Get cluster codebook if any. */
1037  if (n_clust) {
1038  s->mixw_cb = ckd_calloc(1, n_clust);
1039  if (fread(s->mixw_cb, 1, n_clust, fp) != (size_t) n_clust) {
1040  E_ERROR("Failed to read %d bytes from sendump\n", n_clust);
1041  goto error_out;
1042  }
1043  }
1044  }
1045 
1046  /* Set up pointers, or read, or whatever */
1047  if (s->sendump_mmap) {
1048  s->mixw = ckd_calloc_2d(s->n_feat, n_density, sizeof(*s->mixw));
1049  for (n = 0; n < n_feat; n++) {
1050  int step = c;
1051  if (n_bits == 4)
1052  step = (step + 1) / 2;
1053  for (i = 0; i < r; i++) {
1054  s->mixw[n][i] = ((uint8 *) mmio_file_ptr(s->sendump_mmap)) + offset;
1055  offset += step;

1176  }

1185

1188
