135 /* Scans the dictionary and check if all words are present. */
136 static int
137 kws_search_check_dict(kws_search_t * kwss)
138 {
139  dict_t *dict;
140  char **wrdptr;
141  char *tmp_keyphrase;
142  int32 nwrds, wid;
143  int keyword_iter, i;
144  uint8 success;
145 
146  success = TRUE;
147  dict = ps_search_dict(kwss);
148 
149  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
150  tmp_keyphrase = (char *) ckd_salloc(kwss->keyphrases[keyword_iter].word);
151  nwrds = str2words(tmp_keyphrase, NULL, 0);
152  wrdptr = (char **) ckd_calloc(nwrds, sizeof(*wrdptr));
153  str2words(tmp_keyphrase, wrdptr, nwrds);
154  for (i = 0; i < nwrds; i++) {
155  wid = dict_wordid(dict, wrdptr[i]);
156  if (wid == BAD_S3WID) {
157  E_ERROR("The word '%s' is missing in the dictionary\n",
158  wrdptr[i]);
159  success = FALSE;
160  break;
161  }
162  }
163  ckd_free(wrdptr);
164  ckd_free(tmp_keyphrase);
165  }
166  return success;
167 }

169 /* Activate senones for scoring */
170 static void
171 kws_search_sen_active(kws_search_t * kwss)
172 {
173  int i, keyword_iter;
174 
175  acmod_clear_active(ps_search_acmod(kwss));
176 
177  /* active phone loop hmms */
178  for (i = 0; i < kwss->n_pl; i++)
179  acmod_activate_hmm(ps_search_acmod(kwss), &kwss->pl_hmms[i]);
180 
181  /* activate hmms in active nodes */
182  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
183  kws_keyword_t *keyword = &kwss->keyphrases[keyword_iter];
184  for (i = 0; i < keyword->n_hmms; i++) {
185  if (hmm_is_active(keyword->hmms[i]))
186  acmod_activate_hmm(ps_search_acmod(kwss), &keyword->hmms[i]);
187  }
188  }

191 /*
192 * Evaluate all the active HMMs.
193 * (Executed once per frame.)
194 */
195 static void
196 kws_search_hmm_eval(kws_search_t * kwss, int16 const *senscr)
197 {
198  int32 i, keyword_iter;
199  int32 bestscore = WORST_SCORE;
200 
201  hmm_context_set_senscore(kwss->hmmctx, senscr);
202 
203  /* evaluate hmms from phone loop */
204  for (i = 0; i < kwss->n_pl; ++i) {
205  hmm_t *hmm = &kwss->pl_hmms[i];
206  int32 score;
207 
208  score = hmm_vit_eval(hmm);
209  if (score BETTER_THAN bestscore)
210  bestscore = score;
211  }
212  /* evaluate hmms for active nodes */
213  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
214  kws_keyword_t *keyword = &kwss->keyphrases[keyword_iter];
215  for (i = 0; i < keyword->n_hmms; i++) {
216  if (hmm_is_active(keyword->hmms[i])) {
217  hmm_t *hmm = &keyword->hmms[i];
218  int32 score;
219 
220  score = hmm_vit_eval(hmm);
221  if (score BETTER_THAN bestscore)
222  bestscore = score;
223  }
224  }
225  }
226 
227  kwss->bestscore = bestscore;
228 }

230 /*
231 * (Beam) prune the just evaluated HMMs, determine which ones remain
232 * active. Executed once per frame.
233 */
234 static void
235 kws_search_hmm_prune(kws_search_t * kwss)
236 {
237  int32 thresh, i, keyword_iter;
238 
239  thresh = kwss->bestscore + kwss->beam;
240 
241  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
242  kws_keyword_t *keyword = &kwss->keyphrases[keyword_iter];
243  for (i = 0; i < keyword->n_hmms; i++)
244  if (hmm_is_active(keyword->hmms[i]))
245  if (hmm_bestscore(&keyword->hmms[i]) < thresh)
246  hmm_clear(&keyword->hmms[i]);
247  }
248 }

254 static void
255 kws_search_trans(kws_search_t * kwss)
256 {
257  hmm_t *pl_best_hmm = NULL;
258  int32 best_out_score = WORST_SCORE;
259  int i, keyword_iter;
260  uint8 to_clear;
261 
262  /* select best hmm in phone-loop to be a predecessor */
263  for (i = 0; i < kwss->n_pl; i++)
264  if (hmm_out_score(&kwss->pl_hmms[i]) BETTER_THAN best_out_score) {
265  best_out_score = hmm_out_score(&kwss->pl_hmms[i]);
266  pl_best_hmm = &kwss->pl_hmms[i];
267  }
268 
269  /* out probs are not ready yet */
270  if (!pl_best_hmm)
271  return;
272 
273  /* Check whether keyword wasn't spotted yet */
274  to_clear = FALSE;
275  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
276  kws_keyword_t *keyword;
277  hmm_t *last_hmm;
278 
279  keyword = &kwss->keyphrases[keyword_iter];
280  last_hmm = &keyword->hmms[keyword->n_hmms - 1];
281  if (hmm_is_active((*last_hmm))
282  && hmm_out_score(pl_best_hmm) BETTER_THAN WORST_SCORE) {
283 
284  if (hmm_out_score(last_hmm) - hmm_out_score(pl_best_hmm)
285  >= keyword->threshold) {
286 
287  int32 prob = hmm_out_score(last_hmm) - hmm_out_score(pl_best_hmm) -
288  keyword->threshold;
289  kws_detections_add(kwss->detections, keyword->word,
290  hmm_out_history(last_hmm),
291  kwss->frame, prob,
292  hmm_out_score(last_hmm));
293  pl_best_hmm = last_hmm;
294  to_clear = TRUE;
295  } /* keyword is spotted */
296  } /* last hmm of keyword is active */
297  } /* keywords loop */
298 
299  if (to_clear) {
300  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
301  kws_keyword_t* keyword = &kwss->keyphrases[keyword_iter];
302  for (i = 0; i < keyword->n_hmms; i++) {
303  hmm_clear(&keyword->hmms[i]);
304  }
305  }
306  } /* clear all keywords because something was spotted */
307 
308  /* Make transition for all phone loop hmms */
309  for (i = 0; i < kwss->n_pl; i++) {
310  if (hmm_out_score(pl_best_hmm) + kwss->plp BETTER_THAN
311  hmm_in_score(&kwss->pl_hmms[i])) {
312  hmm_enter(&kwss->pl_hmms[i],
313  hmm_out_score(pl_best_hmm) + kwss->plp,
314  hmm_out_history(pl_best_hmm), kwss->frame + 1);
315  }
316  }
317 
318  /* Activate new keyword nodes, enter their hmms */
319  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
320  kws_keyword_t *keyword = &kwss->keyphrases[keyword_iter];
321  for (i = keyword->n_hmms - 1; i > 0; i--) {
322  if (hmm_is_active(&keyword->hmms[i - 1])) {
323  hmm_t *pred_hmm = &keyword->hmms[i - 1];
324 
325  if (!hmm_is_active(&keyword->hmms[i])
326  || hmm_out_score(pred_hmm) BETTER_THAN
327  hmm_in_score(&keyword->hmms[i]))
328  hmm_enter(&keyword->hmms[i], hmm_out_score(pred_hmm),
329  hmm_out_history(pred_hmm), kwss->frame + 1);
330  }
331  }
332 
333  /* Enter keyword start node from phone loop */
334  if (hmm_out_score(pl_best_hmm) BETTER_THAN
335  hmm_in_score(&keyword->hmms[0]))
336  hmm_enter(&keyword->hmms[0], hmm_out_score(pl_best_hmm),
337  kwss->frame, kwss->frame + 1);
338  } /* keywords loop */
339 }

341 static int
342 kws_search_read_list(kws_search_t *kwss, const char* keyfile)
343 {
344  FILE *list_file;
345  lineiter_t *li;
346  int i;
347 
348  if ((list_file = fopen(keyfile, "r")) == NULL) {
349  E_ERROR_SYSTEM("Failed to open keyword file '%s'", keyfile);
350  return -1;
351  }
352 
353  /* count keyphrases amount */
354  kwss->n_keyphrases = 0;
355  for (li = lineiter_start(list_file); li; li = lineiter_next(li))
356  if (li->len > 0)
357  kwss->n_keyphrases++;
358  kwss->keyphrases = (kws_keyword_t *)ckd_calloc(kwss->n_keyphrases, sizeof(*kwss->keyphrases));
359  rewind(list_file);
360 
361  /* read keyphrases */
362  for (li = lineiter_start(list_file), i=0; li; li = lineiter_next(li), i++) {
363  int last_ptr = li->len - 1;
364  kwss->keyphrases[i].threshold = kwss->def_threshold;
365  while (li->buf[last_ptr] == '\n')
366  last_ptr--;
367  if (li->buf[last_ptr] == '/') {
368  int digit_len, start;
369  char digit[16];
370 
371  start = last_ptr - 1;
372  while (li->buf[start] != '/' && start > 0)
373  start--;
374  digit_len = last_ptr - start;
375  memcpy(digit, &li->buf[start+1], digit_len);
376  kwss->keyphrases[i].threshold = (int32) logmath_log(kwss->base.acmod->lmath, atof(digit))
377  >> SENSCR_SHIFT;
378  li->buf[start-1] = '\0';
379  }
380  li->buf[last_ptr + 1] = '\0';
381  kwss->keyphrases[i].word = ckd_salloc(li->buf);
382  }
383 
384  fclose(list_file);
385  return 0;
386 }

388 ps_search_t *
389 kws_search_init(const char *keyphrase,
390  const char *keyfile,
391  cmd_ln_t * config,
392  acmod_t * acmod, dict_t * dict, dict2pid_t * d2p)
393 {
394  kws_search_t *kwss = (kws_search_t *) ckd_calloc(1, sizeof(*kwss));
395  ps_search_init(ps_search_base(kwss), &kws_funcs, config, acmod, dict,
396  d2p);
397 
398  kwss->detections = (kws_detections_t *)ckd_calloc(1, sizeof(*kwss->detections));
399 
400  kwss->beam =
401  (int32) logmath_log(acmod->lmath,
402  cmd_ln_float64_r(config,
403  "-beam")) >> SENSCR_SHIFT;
404 
405  kwss->plp =
406  (int32) logmath_log(acmod->lmath,
407  cmd_ln_float32_r(config,
408  "-kws_plp")) >> SENSCR_SHIFT;
409 
410  kwss->def_threshold =
411  (int32) logmath_log(acmod->lmath,
412  cmd_ln_float64_r(config,
413  "-kws_threshold")) >>
414  SENSCR_SHIFT;
415 
416  E_INFO("KWS(beam: %d, plp: %d, default threshold %d)\n",
417  kwss->beam, kwss->plp, kwss->def_threshold);
418 
419  if (keyfile) {
420  if (kws_search_read_list(kwss, keyfile) < 0) {
421  E_ERROR("Failed to create kws search\n");
422  kws_search_free(ps_search_base(kwss));
423  return NULL;
424  }
425  } else {
426  kwss->n_keyphrases = 1;
427  kwss->keyphrases = (kws_keyword_t *)ckd_calloc(kwss->n_keyphrases, sizeof(*kwss->keyphrases));
428  kwss->keyphrases[0].threshold = kwss->def_threshold;
429  kwss->keyphrases[0].word = ckd_salloc(keyphrase);
430  }
431 
432  /* Check if all words are in dictionary */
433  if (!kws_search_check_dict(kwss)) {
434  kws_search_free(ps_search_base(kwss));
435  return NULL;
436  }
437 
438  /* Reinit for provided keyword */
439  if (kws_search_reinit(ps_search_base(kwss),
440  ps_search_dict(kwss),
441  ps_search_dict2pid(kwss)) < 0) {
442  ps_search_free(ps_search_base(kwss));
443  return NULL;
444  }
445 
446  return ps_search_base(kwss);
447 }

468 int
469 kws_search_reinit(ps_search_t * search, dict_t * dict, dict2pid_t * d2p)
470 {
471  char **wrdptr;
472  char *tmp_keyphrase;
473  int32 wid, pronlen;
474  int32 n_hmms, n_wrds;
475  int32 ssid, tmatid;
476  int i, j, p, keyword_iter;
477  kws_search_t *kwss = (kws_search_t *) search;
478  bin_mdef_t *mdef = search->acmod->mdef;
479  int32 silcipid = bin_mdef_silphone(mdef);
480 
481  /* Free old dict2pid, dict */
482  ps_search_base_reinit(search, dict, d2p);
483 
484  /* Initialize HMM context. */
485  if (kwss->hmmctx)
486  hmm_context_free(kwss->hmmctx);
487  kwss->hmmctx =
488  hmm_context_init(bin_mdef_n_emit_state(search->acmod->mdef),
489  search->acmod->tmat->tp, NULL,
490  search->acmod->mdef->sseq);
491  if (kwss->hmmctx == NULL)
492  return -1;
493 
494  /* Initialize phone loop HMMs. */
495  if (kwss->pl_hmms) {
496  for (i = 0; i < kwss->n_pl; ++i)
497  hmm_deinit((hmm_t *) & kwss->pl_hmms[i]);
498  ckd_free(kwss->pl_hmms);
499  }
500  kwss->n_pl = bin_mdef_n_ciphone(search->acmod->mdef);
501  kwss->pl_hmms =
502  (hmm_t *) ckd_calloc(kwss->n_pl, sizeof(*kwss->pl_hmms));
503  for (i = 0; i < kwss->n_pl; ++i) {
504  hmm_init(kwss->hmmctx, (hmm_t *) & kwss->pl_hmms[i],
505  FALSE,
506  bin_mdef_pid2ssid(search->acmod->mdef, i),
507  bin_mdef_pid2tmatid(search->acmod->mdef, i));
508  }
509 
510  for (keyword_iter = 0; keyword_iter < kwss->n_keyphrases; keyword_iter++) {
511  kws_keyword_t *keyword = &kwss->keyphrases[keyword_iter];
512 
513  /* Initialize keyphrase HMMs */
514  tmp_keyphrase = (char *) ckd_salloc(keyword->word);
515  n_wrds = str2words(tmp_keyphrase, NULL, 0);
516  wrdptr = (char **) ckd_calloc(n_wrds, sizeof(*wrdptr));
517  str2words(tmp_keyphrase, wrdptr, n_wrds);
518 
519  /* count amount of hmms */
520  n_hmms = 0;
521  for (i = 0; i < n_wrds; i++) {
522  wid = dict_wordid(dict, wrdptr[i]);
523  pronlen = dict_pronlen(dict, wid);
524  n_hmms += pronlen;
525  }
526 
527  /* allocate node array */
528  if (keyword->hmms)
529  ckd_free(keyword->hmms);
530  keyword->hmms = (hmm_t *) ckd_calloc(n_hmms, sizeof(hmm_t));
531  keyword->n_hmms = n_hmms;
532 
533  /* fill node array */
534  j = 0;
535  for (i = 0; i < n_wrds; i++) {
536  wid = dict_wordid(dict, wrdptr[i]);
537  pronlen = dict_pronlen(dict, wid);
538  for (p = 0; p < pronlen; p++) {
539  int32 ci = dict_pron(dict, wid, p);
540  if (p == 0) {
541  /* first phone of word */
542  int32 rc =
543  pronlen > 1 ? dict_pron(dict, wid, 1) : silcipid;
544  ssid = dict2pid_ldiph_lc(d2p, ci, rc, silcipid);
545  }
546  else if (p == pronlen - 1) {
547  /* last phone of the word */
548  int32 lc = dict_pron(dict, wid, p - 1);
549  xwdssid_t *rssid = dict2pid_rssid(d2p, ci, lc);
550  int j = rssid->cimap[silcipid];
551  ssid = rssid->ssid[j];
552  }
553  else {
554  /* word internal phone */
555  ssid = dict2pid_internal(d2p, wid, p);
556  }
557  tmatid = bin_mdef_pid2tmatid(mdef, ci);
558  hmm_init(kwss->hmmctx, &keyword->hmms[j], FALSE, ssid,
559  tmatid);
560  j++;
561  }
562  }
563 
564  ckd_free(wrdptr);
565  ckd_free(tmp_keyphrase);
566  }
567 
568  return 0;
569 }

571 int
572 kws_search_start(ps_search_t * search)
573 {
574  int i;
575  kws_search_t *kwss = (kws_search_t *) search;
576 
577  kwss->frame = 0;
578  kwss->bestscore = 0;
579  kws_detections_reset(kwss->detections);
580 
581  /* Reset and enter all phone-loop HMMs. */
582  for (i = 0; i < kwss->n_pl; ++i) {
583  hmm_t *hmm = (hmm_t *) & kwss->pl_hmms[i];
584  hmm_clear(hmm);
585  hmm_enter(hmm, 0, -1, 0);
586  }
587  return 0;
588 }

590 int
591 kws_search_step(ps_search_t * search, int frame_idx)
592 {
593  int16 const *senscr;
594  kws_search_t *kwss = (kws_search_t *) search;
595  acmod_t *acmod = search->acmod;
596 
597  /* Activate senones */
598  if (!acmod->compallsen)
599  kws_search_sen_active(kwss);
600 
601  /* Calculate senone scores for current frame. */
602  senscr = acmod_score(acmod, &frame_idx);
603 
604  /* Evaluate hmms in phone loop and in active keyword nodes */
605  kws_search_hmm_eval(kwss, senscr);
606 
607  /* Prune hmms with low prob */
608  kws_search_hmm_prune(kwss);
609 
610  /* Do hmms transitions */
