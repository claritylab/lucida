130 static void
131 ps_free_searches(ps_decoder_t *ps)
132 {
133  if (ps->searches) {
134  /* Release keys manually as we used ckd_salloc to add them, release every search too. */
135  hash_iter_t *search_it;
136  for (search_it = hash_table_iter(ps->searches); search_it;
137  search_it = hash_table_iter_next(search_it)) {
138  ckd_free((char *) hash_entry_key(search_it->ent));
139  ps_search_free(hash_entry_val(search_it->ent));
140  }
141 
142  hash_table_empty(ps->searches);
143  hash_table_free(ps->searches);
144  }
145 
146  ps->searches = NULL;
147  ps->search = NULL;
148 }

212 int
213 ps_reinit(ps_decoder_t *ps, cmd_ln_t *config)
214 {
215  const char *path;
216  const char *keyphrase;
217  int32 lw;
218 
219  if (config && config != ps->config) {
220  cmd_ln_free_r(ps->config);
221  ps->config = cmd_ln_retain(config);
222  }
223 
224  err_set_debug_level(cmd_ln_int32_r(ps->config, "-debug"));
225  ps->mfclogdir = cmd_ln_str_r(ps->config, "-mfclogdir");
226  ps->rawlogdir = cmd_ln_str_r(ps->config, "-rawlogdir");
227  ps->senlogdir = cmd_ln_str_r(ps->config, "-senlogdir");
228 
229  /* Fill in some default arguments. */
230  ps_init_defaults(ps);
231 
232  /* Free old searches (do this before other reinit) */
233  ps_free_searches(ps);
234  ps->searches = hash_table_new(3, HASH_CASE_YES);
235 
236  /* Free old acmod. */
237  acmod_free(ps->acmod);
238  ps->acmod = NULL;
239 
240  /* Free old dictionary (must be done after the two things above) */
241  dict_free(ps->dict);
242  ps->dict = NULL;
243 
244  /* Free d2p */
245  dict2pid_free(ps->d2p);
246  ps->d2p = NULL;
247 
248  /* Logmath computation (used in acmod and search) */
249  if (ps->lmath == NULL
250  || (logmath_get_base(ps->lmath) !=
251  (float64)cmd_ln_float32_r(ps->config, "-logbase"))) {
252  if (ps->lmath)
253  logmath_free(ps->lmath);
254  ps->lmath = logmath_init
255  ((float64)cmd_ln_float32_r(ps->config, "-logbase"), 0,
256  cmd_ln_boolean_r(ps->config, "-bestpath"));
257  }
258 
259  /* Acoustic model (this is basically everything that
260  * uttproc.c, senscr.c, and others used to do) */
261  if ((ps->acmod = acmod_init(ps->config, ps->lmath, NULL, NULL)) == NULL)
262  return -1;
263 
264  if ((ps->pl_window = cmd_ln_int32_r(ps->config, "-pl_window"))) {
265  /* Initialize an auxiliary phone loop search, which will run in
266  * "parallel" with FSG or N-Gram search. */
267  if ((ps->phone_loop =
268  phone_loop_search_init(ps->config, ps->acmod, ps->dict)) == NULL)
269  return -1;
270  hash_table_enter(ps->searches,
271  ckd_salloc(ps_search_name(ps->phone_loop)),
272  ps->phone_loop);
273  }
274 
275  /* Dictionary and triphone mappings (depends on acmod). */
276  /* FIXME: pass config, change arguments, implement LTS, etc. */
277  if ((ps->dict = dict_init(ps->config, ps->acmod->mdef)) == NULL)
278  return -1;
279  if ((ps->d2p = dict2pid_build(ps->acmod->mdef, ps->dict)) == NULL)
280  return -1;
281 
282  lw = cmd_ln_float32_r(config, "-lw");
283 
284  /* Determine whether we are starting out in FSG or N-Gram search mode.
285  * If neither is used skip search initialization. */
286 
287  /* Load KWS if one was specified in config */
288  if ((keyphrase = cmd_ln_str_r(config, "-keyphrase"))) {
289  if (ps_set_keyphrase(ps, PS_DEFAULT_SEARCH, keyphrase))
290  return -1;
291  ps_set_search(ps, PS_DEFAULT_SEARCH);
292  }
293 
294  if ((path = cmd_ln_str_r(config, "-kws"))) {
295  if (ps_set_kws(ps, PS_DEFAULT_SEARCH, path))
296  return -1;
297  ps_set_search(ps, PS_DEFAULT_SEARCH);
298  }
299 
300  /* Load an FSG if one was specified in config */
301  if ((path = cmd_ln_str_r(config, "-fsg"))) {
302  fsg_model_t *fsg = fsg_model_readfile(path, ps->lmath, lw);
303  if (!fsg)
304  return -1;
305  if (ps_set_fsg(ps, PS_DEFAULT_SEARCH, fsg))
306  return -1;
307  ps_set_search(ps, PS_DEFAULT_SEARCH);
308  }
309 
310  /* Or load a JSGF grammar */
311  if ((path = cmd_ln_str_r(config, "-jsgf"))) {
312  if (ps_set_jsgf_file(ps, PS_DEFAULT_SEARCH, path)
313  || ps_set_search(ps, PS_DEFAULT_SEARCH))
314  return -1;
315  }
316 
317  if ((path = cmd_ln_str_r(ps->config, "-lm"))) {
318  if (ps_set_lm_file(ps, PS_DEFAULT_SEARCH, path)
319  || ps_set_search(ps, PS_DEFAULT_SEARCH))
320  return -1;
321  }
322 
323  if ((path = cmd_ln_str_r(ps->config, "-lmctl"))) {
324  const char *name;
325  ngram_model_t *lmset;
326  ngram_model_set_iter_t *lmset_it;
327 
328  if (!(lmset = ngram_model_set_read(ps->config, path, ps->lmath))) {
329  E_ERROR("Failed to read language model control file: %s\n", path);
330  return -1;
331  }
332 
333  for(lmset_it = ngram_model_set_iter(lmset);
334  lmset_it; lmset_it = ngram_model_set_iter_next(lmset_it)) {
335 
336  ngram_model_t *lm = ngram_model_set_iter_model(lmset_it, &name);
337  E_INFO("adding search %s\n", name);
338  if (ps_set_lm(ps, name, lm)) {
339  ngram_model_free(lm);
340  ngram_model_set_iter_free(lmset_it);
341  return -1;
342  }
343  ngram_model_free(lm);
344  }
345 
346  name = cmd_ln_str_r(config, "-lmname");
347  if (name)
348  ps_set_search(ps, name);
349  else
350  E_WARN("No default LM name (-lmname) for `-lmctl'\n");
351  }
352 
353  /* Initialize performance timer. */
354  ps->perf.name = "decode";
355  ptmr_init(&ps->perf);
356 
357  return 0;
358 }

591 int ps_set_jsgf_file(ps_decoder_t *ps, const char *name, const char *path)
592 {
593  fsg_model_t *fsg;
594  jsgf_rule_t *rule;
595  char const *toprule;
596  jsgf_t *jsgf = jsgf_parse_file(path, NULL);
597  float lw;
598  int result;
599 
600  if (!jsgf)
601  return -1;
602 
603  rule = NULL;
604  /* Take the -toprule if specified. */
605  if ((toprule = cmd_ln_str_r(ps->config, "-toprule"))) {
606  char *ruletok;
607  ruletok = string_join("<", toprule, ">", NULL);
608  rule = jsgf_get_rule(jsgf, ruletok);
609  ckd_free(ruletok);
610  if (rule == NULL) {
611  E_ERROR("Start rule %s not found\n", toprule);
612  return -1;
613  }
614  } else {
615  /* Otherwise, take the first public rule. */
616  jsgf_rule_iter_t *itor;
617 
618  for (itor = jsgf_rule_iter(jsgf); itor;
619  itor = jsgf_rule_iter_next(itor)) {
620  rule = jsgf_rule_iter_rule(itor);
621  if (jsgf_rule_public(rule)) {
622  jsgf_rule_iter_free(itor);
623  break;
624  }
625  }
626  if (rule == NULL) {
627  E_ERROR("No public rules found in %s\n", path);
628  return -1;
629  }
630  }
631 
632  lw = cmd_ln_float32_r(ps->config, "-lw");
633  fsg = jsgf_build_fsg(jsgf, rule, ps->lmath, lw);

772  char const *uttid, long maxsamps)
773 {

775

777  /* If this file is seekable or maxsamps is specified, then decode
778  * the whole thing at once. */
779  if (maxsamps != -1 || (pos = ftell(rawfh)) >= 0) {
780  int16 *data;

782  if (maxsamps == -1) {
783  long endpos;
784  fseek(rawfh, 0, SEEK_END);

786  fseek(rawfh, pos, SEEK_SET);
787  maxsamps = endpos - pos;

790  total = fread(data, sizeof(*data), maxsamps, rawfh);
791  ps_process_raw(ps, data, total, FALSE, TRUE);

796  total = 0;
797  while (!feof(rawfh)) {

800

802  ps_process_raw(ps, data, nread, FALSE, FALSE);
803  total += nread;
804  }
805  }
806  ps_end_utt(ps);

1004 int
1005 ps_end_utt(ps_decoder_t *ps)
1006 {
1007  int rv, i;
1008 
1009  acmod_end_utt(ps->acmod);
1010 
1011  /* Search any remaining frames. */
1012  if ((rv = ps_search_forward(ps)) < 0) {
1013  ptmr_stop(&ps->perf);
1014  return rv;
1015  }
1016  /* Finish phone loop search. */
1017  if (ps->phone_loop) {
1018  if ((rv = ps_search_finish(ps->phone_loop)) < 0) {
1019  ptmr_stop(&ps->perf);
1020  return rv;
1021  }
1022  }
1023  /* Search any frames remaining in the lookahead window. */
1024  for (i = ps->acmod->output_frame - ps->pl_window;
1025  i < ps->acmod->output_frame; ++i)
1026  ps_search_step(ps->search, i);
1027  /* Finish main search. */
1028  if ((rv = ps_search_finish(ps->search)) < 0) {
1029  ptmr_stop(&ps->perf);
1030  return rv;
1031  }
1032  ptmr_stop(&ps->perf);
1033 
1034  /* Log a backtrace if requested. */
1035  if (cmd_ln_boolean_r(ps->config, "-backtrace")) {
1036  char const *uttid, *hyp;
1037  ps_seg_t *seg;
1038  int32 score;
1039 
1040  hyp = ps_get_hyp(ps, &score, &uttid);
1041  E_INFO("%s: %s (%d)\n", uttid, hyp, score);
1042  E_INFO_NOFN("%-20s %-5s %-5s %-5s %-10s %-10s %-3s\n",
1043  "word", "start", "end", "pprob", "ascr", "lscr", "lback");
1044  for (seg = ps_seg_iter(ps, &score); seg;
1045  seg = ps_seg_next(seg)) {

1048  int32 post, lscr, ascr, lback;
1049 
1050  word = ps_seg_word(seg);
1051  ps_seg_frames(seg, &sf, &ef);

1054  word, sf, ef, logmath_exp(ps_get_logmath(ps), post),
1055  ascr, lscr, lback);
1056  }
1057  }
1058  return rv;
1059 }
