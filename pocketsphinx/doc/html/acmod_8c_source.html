77 static int
78 acmod_init_am(acmod_t *acmod)
79 {
80  char const *mdeffn, *tmatfn, *mllrfn, *hmmdir;
81 
82  /* Read model definition. */
83  if ((mdeffn = cmd_ln_str_r(acmod->config, "-mdef")) == NULL) {
84  if ((hmmdir = cmd_ln_str_r(acmod->config, "-hmm")) == NULL)
85  E_ERROR("Acoustic model definition is not specified either "
86  "with -mdef option or with -hmm\n");
87  else
88  E_ERROR("Folder '%s' does not contain acoustic model "
89  "definition 'mdef'\n", hmmdir);
90 
91  return -1;
92  }
93 
94  if ((acmod->mdef = bin_mdef_read(acmod->config, mdeffn)) == NULL) {
95  E_ERROR("Failed to read acoustic model definition from %s\n", mdeffn);
96  return -1;
97  }
98 
99  /* Read transition matrices. */
100  if ((tmatfn = cmd_ln_str_r(acmod->config, "-tmat")) == NULL) {
101  E_ERROR("No tmat file specified\n");
102  return -1;
103  }
104  acmod->tmat = tmat_init(tmatfn, acmod->lmath,
105  cmd_ln_float32_r(acmod->config, "-tmatfloor"),
106  TRUE);
107 
108  /* Read the acoustic models. */
109  if ((cmd_ln_str_r(acmod->config, "-mean") == NULL)
110  || (cmd_ln_str_r(acmod->config, "-var") == NULL)
111  || (cmd_ln_str_r(acmod->config, "-tmat") == NULL)) {
112  E_ERROR("No mean/var/tmat files specified\n");
113  return -1;
114  }
115 
116  if (cmd_ln_str_r(acmod->config, "-senmgau")) {
117  E_INFO("Using general multi-stream GMM computation\n");
118  acmod->mgau = ms_mgau_init(acmod, acmod->lmath, acmod->mdef);
119  if (acmod->mgau == NULL)
120  return -1;
121  }
122  else {
123  E_INFO("Attempting to use SCHMM computation module\n");
124  if ((acmod->mgau = s2_semi_mgau_init(acmod)) == NULL) {
125  E_INFO("Attempting to use PTHMM computation module\n");
126  if ((acmod->mgau = ptm_mgau_init(acmod, acmod->mdef)) == NULL) {
127  E_INFO("Falling back to general multi-stream GMM computation\n");
128  acmod->mgau = ms_mgau_init(acmod, acmod->lmath, acmod->mdef);
129  if (acmod->mgau == NULL)
130  return -1;
131  }
132  }
133  }
134 
135  /* If there is an MLLR transform, apply it. */
136  if ((mllrfn = cmd_ln_str_r(acmod->config, "-mllr"))) {
137  ps_mllr_t *mllr = ps_mllr_read(mllrfn);
138  if (mllr == NULL)
139  return -1;
140  acmod_update_mllr(acmod, mllr);
141  }
142 
143  return 0;
144 }

146 static int
147 acmod_init_feat(acmod_t *acmod)
148 {
149  acmod->fcb =
150  feat_init(cmd_ln_str_r(acmod->config, "-feat"),
151  cmn_type_from_str(cmd_ln_str_r(acmod->config,"-cmn")),
152  cmd_ln_boolean_r(acmod->config, "-varnorm"),
153  agc_type_from_str(cmd_ln_str_r(acmod->config, "-agc")),
154  1, cmd_ln_int32_r(acmod->config, "-ceplen"));
155  if (acmod->fcb == NULL)
156  return -1;
157 
158  if (cmd_ln_str_r(acmod->config, "-lda")) {
159  E_INFO("Reading linear feature transformation from %s\n",
160  cmd_ln_str_r(acmod->config, "-lda"));
161  if (feat_read_lda(acmod->fcb,
162  cmd_ln_str_r(acmod->config, "-lda"),
163  cmd_ln_int32_r(acmod->config, "-ldadim")) < 0)
164  return -1;
165  }
166 
167  if (cmd_ln_str_r(acmod->config, "-svspec")) {
168  int32 **subvecs;
169  E_INFO("Using subvector specification %s\n",
170  cmd_ln_str_r(acmod->config, "-svspec"));
171  if ((subvecs = parse_subvecs(cmd_ln_str_r(acmod->config, "-svspec"))) == NULL)
172  return -1;
173  if ((feat_set_subvecs(acmod->fcb, subvecs)) < 0)
174  return -1;
175  }
176 
177  if (cmd_ln_exists_r(acmod->config, "-agcthresh")
178  && 0 != strcmp(cmd_ln_str_r(acmod->config, "-agc"), "none")) {
179  agc_set_threshold(acmod->fcb->agc_struct,
180  cmd_ln_float32_r(acmod->config, "-agcthresh"));
181  }
182 
183  if (acmod->fcb->cmn_struct
184  && cmd_ln_exists_r(acmod->config, "-cmninit")) {
185  char *c, *cc, *vallist;
186  int32 nvals;
187 
188  vallist = ckd_salloc(cmd_ln_str_r(acmod->config, "-cmninit"));
189  c = vallist;
190  nvals = 0;
191  while (nvals < acmod->fcb->cmn_struct->veclen
192  && (cc = strchr(c, ',')) != NULL) {
193  *cc = '\0';
194  acmod->fcb->cmn_struct->cmn_mean[nvals] = FLOAT2MFCC(atof(c));
195  c = cc + 1;
196  ++nvals;
197  }
198  if (nvals < acmod->fcb->cmn_struct->veclen && *c != '\0') {
199  acmod->fcb->cmn_struct->cmn_mean[nvals] = FLOAT2MFCC(atof(c));
200  }
201  ckd_free(vallist);
202  }
203  return 0;
204 }

235 acmod_t *
236 acmod_init(cmd_ln_t *config, logmath_t *lmath, fe_t *fe, feat_t *fcb)
237 {
238  acmod_t *acmod;
239  char const *featparams;
240 
241  acmod = ckd_calloc(1, sizeof(*acmod));
242  acmod->config = cmd_ln_retain(config);
243  acmod->lmath = lmath;
244  acmod->state = ACMOD_IDLE;
245 
246  /* Look for feat.params in acoustic model dir. */
247  if ((featparams = cmd_ln_str_r(acmod->config, "-featparams"))) {
248  if (NULL !=
249  cmd_ln_parse_file_r(acmod->config, feat_defn, featparams, FALSE))
250  E_INFO("Parsed model-specific feature parameters from %s\n",
251  featparams);
252  }
253 
254  /* Initialize feature computation. */
255  if (fe) {
256  if (acmod_fe_mismatch(acmod, fe))
257  goto error_out;
258  fe_retain(fe);
259  acmod->fe = fe;
260  }
261  else {
262  /* Initialize a new front end. */
263  acmod->fe = fe_init_auto_r(config);
264  if (acmod->fe == NULL)
265  goto error_out;
266  if (acmod_fe_mismatch(acmod, acmod->fe))
267  goto error_out;
268  }
269  if (fcb) {
270  if (acmod_feat_mismatch(acmod, fcb))
271  goto error_out;
272  feat_retain(fcb);
273  acmod->fcb = fcb;
274  }
275  else {
276  /* Initialize a new fcb. */
277  if (acmod_init_feat(acmod) < 0)
278  goto error_out;
279  }
280 
281  /* Load acoustic model parameters. */
282  if (acmod_init_am(acmod) < 0)
283  goto error_out;
284 
285 
286  /* The MFCC buffer needs to be at least as large as the dynamic
287  * feature window. */
288  acmod->n_mfc_alloc = acmod->fcb->window_size * 2 + 1;
289  acmod->mfc_buf = (mfcc_t **)
290  ckd_calloc_2d(acmod->n_mfc_alloc, acmod->fcb->cepsize,
291  sizeof(**acmod->mfc_buf));
292 
293  /* Feature buffer has to be at least as large as MFCC buffer. */
294  acmod->n_feat_alloc = acmod->n_mfc_alloc + cmd_ln_int32_r(config, "-pl_window");
295  acmod->feat_buf = feat_array_alloc(acmod->fcb, acmod->n_feat_alloc);
296  acmod->framepos = ckd_calloc(acmod->n_feat_alloc, sizeof(*acmod->framepos));
297 
298  /* Senone computation stuff. */
299  acmod->senone_scores = ckd_calloc(bin_mdef_n_sen(acmod->mdef),
300  sizeof(*acmod->senone_scores));
301  acmod->senone_active_vec = bitvec_alloc(bin_mdef_n_sen(acmod->mdef));
302  acmod->senone_active = ckd_calloc(bin_mdef_n_sen(acmod->mdef),
303  sizeof(*acmod->senone_active));
304  acmod->log_zero = logmath_get_zero(acmod->lmath);
305  acmod->compallsen = cmd_ln_boolean_r(config, "-compallsen");
306  return acmod;
307 
308 error_out:
309  acmod_free(acmod);
310  return NULL;
311 }

452 int
453 acmod_end_utt(acmod_t *acmod)
454 {
455  int32 nfr = 0;
456 
457  acmod->state = ACMOD_ENDED;
458  if (acmod->n_mfc_frame < acmod->n_mfc_alloc) {
459  int inptr;
460  /* Where to start writing them (circular buffer) */
461  inptr = (acmod->mfc_outidx + acmod->n_mfc_frame) % acmod->n_mfc_alloc;
462  /* nfr is always either zero or one. */
463  fe_end_utt(acmod->fe, acmod->mfc_buf[inptr], &nfr);
464  acmod->n_mfc_frame += nfr;
465  /* Process whatever's left, and any leadout. */
466  nfr = acmod_process_mfcbuf(acmod);
467  }
468  if (acmod->mfcfh) {
469  int32 outlen, rv;
470  outlen = (ftell(acmod->mfcfh) - 4) / 4;
471  if (!WORDS_BIGENDIAN)
472  SWAP_INT32(&outlen);
473  /* Try to seek and write */
474  if ((rv = fseek(acmod->mfcfh, 0, SEEK_SET)) == 0) {
475  fwrite(&outlen, 4, 1, acmod->mfcfh);
476  }
477  fclose(acmod->mfcfh);
478  acmod->mfcfh = NULL;
479  }
480  if (acmod->rawfh) {
481  fclose(acmod->rawfh);
482  acmod->rawfh = NULL;
483  }
484 
485  if (acmod->senfh) {
486  fclose(acmod->senfh);
487  acmod->senfh = NULL;
488  }
489 
490  return nfr;
491 }

493 static int
494 acmod_log_mfc(acmod_t *acmod,
495  mfcc_t **cep, int n_frames)
496 {
497  int i, n;
498  int32 *ptr = (int32 *)cep[0];
499 
500  n = n_frames * feat_cepsize(acmod->fcb);
501  /* Swap bytes. */
502  if (!WORDS_BIGENDIAN) {
503  for (i = 0; i < (n * sizeof(mfcc_t)); ++i) {
504  SWAP_INT32(ptr + i);
505  }
506  }
507  /* Write features. */
508  if (fwrite(cep[0], sizeof(mfcc_t), n, acmod->mfcfh) != n) {
509  E_ERROR_SYSTEM("Failed to write %d values to log file", n);
510  }
511 
512  /* Swap them back. */
513  if (!WORDS_BIGENDIAN) {
514  for (i = 0; i < (n * sizeof(mfcc_t)); ++i) {
515  SWAP_INT32(ptr + i);
516  }
517  }
518  return 0;
519 }

521 static int
522 acmod_process_full_cep(acmod_t *acmod,
523  mfcc_t ***inout_cep,
524  int *inout_n_frames)
525 {
526  int32 nfr;
527 
528  /* Write to log file. */
529  if (acmod->mfcfh)
530  acmod_log_mfc(acmod, *inout_cep, *inout_n_frames);
531 
532  /* Resize feat_buf to fit. */
533  if (acmod->n_feat_alloc < *inout_n_frames) {
534 
535  if (*inout_n_frames > MAX_N_FRAMES)
536  E_FATAL("Batch processing can not process more than %d frames "
537  "at once, requested %d\n", MAX_N_FRAMES, *inout_n_frames);
538 
539  feat_array_free(acmod->feat_buf);
540  acmod->feat_buf = feat_array_alloc(acmod->fcb, *inout_n_frames);
541  acmod->n_feat_alloc = *inout_n_frames;
542  acmod->n_feat_frame = 0;
543  acmod->feat_outidx = 0;
544  }
545  /* Make dynamic features. */
546  nfr = feat_s2mfc2feat_live(acmod->fcb, *inout_cep, inout_n_frames,
547  TRUE, TRUE, acmod->feat_buf);
548  acmod->n_feat_frame = nfr;
549  assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
550  *inout_cep += *inout_n_frames;
551  *inout_n_frames = 0;
552 
553  return nfr;
554 }

556 static int
557 acmod_process_full_raw(acmod_t *acmod,
558  int16 const **inout_raw,
559  size_t *inout_n_samps)
560 {
561  int32 nfr, ntail;
562  mfcc_t **cepptr;
563 
564  /* Write to logging file if any. */
565  if (acmod->rawfh)
566  fwrite(*inout_raw, 2, *inout_n_samps, acmod->rawfh);
567  /* Resize mfc_buf to fit. */
568  if (fe_process_frames(acmod->fe, NULL, inout_n_samps, NULL, &nfr) < 0)
569  return -1;
570  if (acmod->n_mfc_alloc < nfr + 1) {
571  ckd_free_2d(acmod->mfc_buf);
572  acmod->mfc_buf = ckd_calloc_2d(nfr + 1, fe_get_output_size(acmod->fe),
573  sizeof(**acmod->mfc_buf));
574  acmod->n_mfc_alloc = nfr + 1;
575  }
576  acmod->n_mfc_frame = 0;
577  acmod->mfc_outidx = 0;
578  fe_start_utt(acmod->fe);
579  if (fe_process_frames(acmod->fe, inout_raw, inout_n_samps,
580  acmod->mfc_buf, &nfr) < 0)
581  return -1;
582  fe_end_utt(acmod->fe, acmod->mfc_buf[nfr], &ntail);
583  nfr += ntail;
584 
585  cepptr = acmod->mfc_buf;
586  nfr = acmod_process_full_cep(acmod, &cepptr, &nfr);
587  acmod->n_mfc_frame = 0;
588  return nfr;
589 }

594 static int32
595 acmod_process_mfcbuf(acmod_t *acmod)
596 {
597  mfcc_t **mfcptr;
598  int32 ncep;
599 
600  ncep = acmod->n_mfc_frame;
601  /* Also do this in two parts because of the circular mfc_buf. */
602  if (acmod->mfc_outidx + ncep > acmod->n_mfc_alloc) {
603  int32 ncep1 = acmod->n_mfc_alloc - acmod->mfc_outidx;
604  int saved_state = acmod->state;
605 
606  /* Make sure we don't end the utterance here. */
607  if (acmod->state == ACMOD_ENDED)
608  acmod->state = ACMOD_PROCESSING;
609  mfcptr = acmod->mfc_buf + acmod->mfc_outidx;
610  ncep1 = acmod_process_cep(acmod, &mfcptr, &ncep1, FALSE);
611  /* It's possible that not all available frames were filled. */
612  ncep -= ncep1;
613  acmod->n_mfc_frame -= ncep1;
614  acmod->mfc_outidx += ncep1;
615  acmod->mfc_outidx %= acmod->n_mfc_alloc;
616  /* Restore original state (could this really be the end) */
617  acmod->state = saved_state;
618  }
619  mfcptr = acmod->mfc_buf + acmod->mfc_outidx;
620  ncep = acmod_process_cep(acmod, &mfcptr, &ncep, FALSE);
621  acmod->n_mfc_frame -= ncep;
622  acmod->mfc_outidx += ncep;
623  acmod->mfc_outidx %= acmod->n_mfc_alloc;
624  return ncep;
625 }

627 int
628 acmod_process_raw(acmod_t *acmod,
629  int16 const **inout_raw,
630  size_t *inout_n_samps,
631  int full_utt)
632 {
633  int32 ncep;
634 
635  /* If this is a full utterance, process it all at once. */
636  if (full_utt)
637  return acmod_process_full_raw(acmod, inout_raw, inout_n_samps);
638 
639  /* Append MFCCs to the end of any that are previously in there
640  * (in practice, there will probably be none) */
641  if (inout_n_samps && *inout_n_samps) {
642  int16 const *prev_audio_inptr = *inout_raw;
643  int inptr;
644 
645  /* Total number of frames available. */
646  ncep = acmod->n_mfc_alloc - acmod->n_mfc_frame;
647  /* Where to start writing them (circular buffer) */
648  inptr = (acmod->mfc_outidx + acmod->n_mfc_frame) % acmod->n_mfc_alloc;
649 
650  /* Write them in two (or more) parts if there is wraparound. */
651  while (inptr + ncep > acmod->n_mfc_alloc) {
652  int32 ncep1 = acmod->n_mfc_alloc - inptr;
653  if (fe_process_frames(acmod->fe, inout_raw, inout_n_samps,
654  acmod->mfc_buf + inptr, &ncep1) < 0)
655  return -1;
656  /* Write to logging file if any. */
657  if (acmod->rawfh) {
658  fwrite(prev_audio_inptr, 2,
659  *inout_raw - prev_audio_inptr,
660  acmod->rawfh);
661  prev_audio_inptr = *inout_raw;
662  }
663  /* ncep1 now contains the number of frames actually
664  * processed. This is a good thing, but it means we
665  * actually still might have some room left at the end of
666  * the buffer, hence the while loop. Unfortunately it
667  * also means that in the case where we are really
668  * actually done, we need to get out totally, hence the
669  * goto. */
670  acmod->n_mfc_frame += ncep1;
671  ncep -= ncep1;
672  inptr += ncep1;
673  inptr %= acmod->n_mfc_alloc;
674  if (ncep1 == 0)
675  goto alldone;
676  }
677  assert(inptr + ncep <= acmod->n_mfc_alloc);
678  if (fe_process_frames(acmod->fe, inout_raw, inout_n_samps,
679  acmod->mfc_buf + inptr, &ncep) < 0)
680  return -1;
681  /* Write to logging file if any. */
682  if (acmod->rawfh) {
683  fwrite(prev_audio_inptr, 2,
684  *inout_raw - prev_audio_inptr, acmod->rawfh);
685  prev_audio_inptr = *inout_raw;
686  }
687  acmod->n_mfc_frame += ncep;
688  alldone:
689  ;
690  }
691 
692  /* Hand things off to acmod_process_cep. */
693  return acmod_process_mfcbuf(acmod);
694 }

696 int
697 acmod_process_cep(acmod_t *acmod,
698  mfcc_t ***inout_cep,
699  int *inout_n_frames,
700  int full_utt)
701 {
702  int32 nfeat, ncep, inptr;
703  int orig_n_frames;
704 
705  /* If this is a full utterance, process it all at once. */
706  if (full_utt)
707  return acmod_process_full_cep(acmod, inout_cep, inout_n_frames);
708 
709  /* Write to log file. */
710  if (acmod->mfcfh)
711  acmod_log_mfc(acmod, *inout_cep, *inout_n_frames);
712 
713  /* Maximum number of frames we're going to generate. */
714  orig_n_frames = ncep = nfeat = *inout_n_frames;
715 
716  /* FIXME: This behaviour isn't guaranteed... */
717  if (acmod->state == ACMOD_ENDED)
718  nfeat += feat_window_size(acmod->fcb);
719  else if (acmod->state == ACMOD_STARTED)
720  nfeat -= feat_window_size(acmod->fcb);
721 
722  /* Clamp number of features to fit available space. */
723  if (nfeat > acmod->n_feat_alloc - acmod->n_feat_frame) {
724  /* Grow it as needed - we have to grow it at the end of an
725  * utterance because we can't return a short read there. */
726  if (acmod->grow_feat || acmod->state == ACMOD_ENDED)
727  acmod_grow_feat_buf(acmod, acmod->n_feat_alloc + nfeat);
728  else
729  ncep -= (nfeat - (acmod->n_feat_alloc - acmod->n_feat_frame));
730  }
731 
732  /* Where to start writing in the feature buffer. */
733  if (acmod->grow_feat) {
734  /* Grow to avoid wraparound if grow_feat == TRUE. */
735  inptr = acmod->feat_outidx + acmod->n_feat_frame;
736  while (inptr + nfeat >= acmod->n_feat_alloc)
737  acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
738  }
739  else {
740  inptr = (acmod->feat_outidx + acmod->n_feat_frame) % acmod->n_feat_alloc;
741  }
742 
743 
744  /* FIXME: we can't split the last frame drop properly to be on the bounary,
745  * so just return
746  */
747  if (inptr + nfeat > acmod->n_feat_alloc && acmod->state == ACMOD_ENDED) {
748  *inout_n_frames -= ncep;
749  *inout_cep += ncep;
750  return 0;
751  }
752 
753  /* Write them in two parts if there is wraparound. */
754  if (inptr + nfeat > acmod->n_feat_alloc) {
755  int32 ncep1 = acmod->n_feat_alloc - inptr;
756 
757  /* Make sure we don't end the utterance here. */
758  nfeat = feat_s2mfc2feat_live(acmod->fcb, *inout_cep,
759  &ncep1,
760  (acmod->state == ACMOD_STARTED),
761  FALSE,
762  acmod->feat_buf + inptr);
763  if (nfeat < 0)
764  return -1;
765  /* Move the output feature pointer forward. */
766  acmod->n_feat_frame += nfeat;
767  assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
768  inptr += nfeat;
769  inptr %= acmod->n_feat_alloc;
770  /* Move the input feature pointers forward. */
771  *inout_n_frames -= ncep1;
772  *inout_cep += ncep1;
773  ncep -= ncep1;
774  }
775 
776  nfeat = feat_s2mfc2feat_live(acmod->fcb, *inout_cep,
777  &ncep,
778  (acmod->state == ACMOD_STARTED),
779  (acmod->state == ACMOD_ENDED),
780  acmod->feat_buf + inptr);
781  if (nfeat < 0)
782  return -1;
783  acmod->n_feat_frame += nfeat;
784  assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
785  /* Move the input feature pointers forward. */
786  *inout_n_frames -= ncep;
787  *inout_cep += ncep;
788  if (acmod->state == ACMOD_STARTED)
789  acmod->state = ACMOD_PROCESSING;
790  return orig_n_frames - *inout_n_frames;
791 }

793 int
794 acmod_process_feat(acmod_t *acmod,
795  mfcc_t **feat)
796 {
797  int i, inptr;
798 
799  if (acmod->n_feat_frame == acmod->n_feat_alloc) {
800  if (acmod->grow_feat)
801  acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
802  else
803  return 0;
804  }
805 
806  if (acmod->grow_feat) {
807  /* Grow to avoid wraparound if grow_feat == TRUE. */
808  inptr = acmod->feat_outidx + acmod->n_feat_frame;
809  while (inptr + 1 >= acmod->n_feat_alloc)
810  acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
811  }
812  else {
813  inptr = (acmod->feat_outidx + acmod->n_feat_frame) % acmod->n_feat_alloc;
814  }
815  for (i = 0; i < feat_dimension1(acmod->fcb); ++i)
816  memcpy(acmod->feat_buf[inptr][i],
817  feat[i], feat_dimension2(acmod->fcb, i) * sizeof(**feat));
818  ++acmod->n_feat_frame;
819  assert(acmod->n_feat_frame <= acmod->n_feat_alloc);
820 
821  return 1;
822 }

907 int
908 acmod_write_scores(acmod_t *acmod, int n_active, uint8 const *active,
909  int16 const *senscr, FILE *senfh)
910 {
911  int16 n_active2;
912 
913  /* Uncompressed frame format:
914  *
915  * (2 bytes) n_active: Number of active senones
916  * If all senones active:
917  * (n_active * 2 bytes) scores of active senones
918  *
919  * Otherwise:
920  * (2 bytes) n_active: Number of active senones
921  * (n_active bytes) deltas to active senones
922  * (n_active * 2 bytes) scores of active senones
923  */
924  n_active2 = n_active;
925  if (fwrite(&n_active2, 2, 1, senfh) != 1)
926  goto error_out;
927  if (n_active == bin_mdef_n_sen(acmod->mdef)) {
928  if (fwrite(senscr, 2, n_active, senfh) != n_active)
929  goto error_out;
930  }
931  else {
932  int i, n;
933  if (fwrite(active, 1, n_active, senfh) != n_active)
934  goto error_out;
935  for (i = n = 0; i < n_active; ++i) {
936  n += active[i];
937  if (fwrite(senscr + n, 2, 1, senfh) != 1)
938  goto error_out;
939  }
940  }
941  return 0;
942 error_out:
943  E_ERROR_SYSTEM("Failed to write frame to senone file");
944  return -1;
945 }

950 static int
951 acmod_read_scores_internal(acmod_t *acmod)
952 {
953  FILE *senfh = acmod->insenfh;
954  int16 n_active;
955  int rv;
956 
957  if (acmod->n_feat_frame == acmod->n_feat_alloc) {
958  if (acmod->grow_feat)
959  acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
960  else
961  return 0;
962  }
963 
964  if (senfh == NULL)
965  return -1;
966  if ((rv = fread(&n_active, 2, 1, senfh)) < 0)
967  goto error_out;
968  else if (rv == 0)
969  return 0;
970 
971  acmod->n_senone_active = n_active;
972  if (acmod->n_senone_active == bin_mdef_n_sen(acmod->mdef)) {
973  if ((rv = fread(acmod->senone_scores, 2,
974  acmod->n_senone_active, senfh)) < 0)
975  goto error_out;
976  else if (rv != acmod->n_senone_active)
977  return 0;
978  }
979  else {
980  int i, n;
981  if ((rv = fread(acmod->senone_active, 1,
982  acmod->n_senone_active, senfh)) < 0)
983  goto error_out;
984  else if (rv != acmod->n_senone_active)
985  return 0;
986  for (i = 0, n = 0; i < acmod->n_senone_active; ++i) {
987  int j, sen = n + acmod->senone_active[i];
988  for (j = n + 1; j < sen; ++j)
989  acmod->senone_scores[j] = SENSCR_DUMMY;
990  if ((rv = fread(acmod->senone_scores + sen, 2, 1, senfh)) < 0)
991  goto error_out;
992  else if (rv == 0)
993  return 0;
994  n = sen;
995  }
996  ++n;
997  while (n < bin_mdef_n_sen(acmod->mdef))
998  acmod->senone_scores[n++] = SENSCR_DUMMY;
999  }
1000  return 1;
1001 error_out:
1002  E_ERROR_SYSTEM("Failed to read frame from senone file");
1003  return -1;
1004 }

1006 int
1007 acmod_read_scores(acmod_t *acmod)
1008 {
1009  int inptr, rv;
1010 
1011  if (acmod->grow_feat) {
1012  /* Grow to avoid wraparound if grow_feat == TRUE. */
1013  inptr = acmod->feat_outidx + acmod->n_feat_frame;
1014  /* Has to be +1, otherwise, next time acmod_advance() is
1015  * called, this will wrap around. */
1016  while (inptr + 1 >= acmod->n_feat_alloc)
1017  acmod_grow_feat_buf(acmod, acmod->n_feat_alloc * 2);
1018  }
1019  else {
1020  inptr = (acmod->feat_outidx + acmod->n_feat_frame) %
1021  acmod->n_feat_alloc;
1022  }
1023 
1024  if ((rv = acmod_read_scores_internal(acmod)) != 1)
1025  return rv;
1026 
1027  /* Set acmod->senscr_frame appropriately so that these scores
1028  get reused below in acmod_score(). */
1029  acmod->senscr_frame = acmod->output_frame + acmod->n_feat_frame;
1030 
1031  E_DEBUG(1,("Frame %d has %d active states\n",
1032  acmod->senscr_frame, acmod->n_senone_active));
1033 
1034  /* Increment the "feature frame counter" and record the file
1035  * position for the relevant frame in the (possibly circular)
1036  * buffer. */
1037  ++acmod->n_feat_frame;
1038  acmod->framepos[inptr] = ftell(acmod->insenfh);

1100 int16 const *
1101 acmod_score(acmod_t *acmod, int *inout_frame_idx)
1102 {
1103  int frame_idx, feat_idx;
1104 
1105  /* Calculate the absolute frame index to be scored. */
1106  frame_idx = calc_frame_idx(acmod, inout_frame_idx);
1107 
1108  /* If all senones are being computed, or we are using a senone file,
1109  then we can reuse existing scores. */
1110  if ((acmod->compallsen || acmod->insenfh)
1111  && frame_idx == acmod->senscr_frame) {
1112  if (inout_frame_idx)
1113  *inout_frame_idx = frame_idx;
1114  return acmod->senone_scores;
1115  }
1116 
1117  /* Calculate position of requested frame in circular buffer. */
1118  if ((feat_idx = calc_feat_idx(acmod, frame_idx)) < 0)
1119  return NULL;
1120 
1121  /*
1122  * If there is an input senone file locate the appropriate frame and read
1123  * it.
1124  */
1125  if (acmod->insenfh) {
1126  fseek(acmod->insenfh, acmod->framepos[feat_idx], SEEK_SET);
1127  if (acmod_read_scores_internal(acmod) < 0)
1128  return NULL;
1129  }
1130  else {
1131  /* Build active senone list. */
1132  acmod_flags2list(acmod);
1133 
1134  /* Generate scores for the next available frame */
1135  ps_mgau_frame_eval(acmod->mgau,
1136  acmod->senone_scores,
1137  acmod->senone_active,
1138  acmod->n_senone_active,
1139  acmod->feat_buf[feat_idx],
1140  frame_idx,
1141  acmod->compallsen);
1142  }
1143 
1144  if (inout_frame_idx)
1145  *inout_frame_idx = frame_idx;
1146  acmod->senscr_frame = frame_idx;
1147 
1148  /* Dump scores to the senone dump file if one exists. */
1149  if (acmod->senfh) {
1150  if (acmod_write_scores(acmod, acmod->n_senone_active,
1151  acmod->senone_active,
1152  acmod->senone_scores,
1153  acmod->senfh) < 0)
1154  return NULL;
1155  E_DEBUG(1,("Frame %d has %d active states\n", frame_idx,
1156  acmod->n_senone_active));
1157  }
1158 
1159  return acmod->senone_scores;
1160 }
