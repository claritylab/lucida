240 {

242  int old_n_words;

244 
245  /* Update the number of words. */
246  old_n_words = search->n_words;
247  if (old_n_words != dict_size(dict)) {
248  search->n_words = dict_size(dict);
249  /* Reallocate these temporary arrays. */
250  ckd_free(ngs->word_lat_idx);
251  ckd_free(ngs->word_active);
252  ckd_free(ngs->last_ltrans);
253  ckd_free_2d(ngs->active_word_list);
254  ngs->word_lat_idx = ckd_calloc(search->n_words, sizeof(*ngs->word_lat_idx));
255  ngs->word_active = bitvec_alloc(search->n_words);
256  ngs->last_ltrans = ckd_calloc(search->n_words, sizeof(*ngs->last_ltrans));
257  ngs->active_word_list
258  = ckd_calloc_2d(2, search->n_words,
259  sizeof(**ngs->active_word_list));
260  }
261 
262  /* Free old dict2pid, dict */
263  ps_search_base_reinit(search, dict, d2p);
264 
265  if (ngs->lmset == NULL)
266  return 0;
267 
268  /* Update beam widths. */
269  ngram_search_calc_beams(ngs);
270 
271  /* Update word mappings. */
272  ngram_search_update_widmap(ngs);
273 
274  /* Now rebuild lextrees. */
275  if (ngs->fwdtree) {
276  if ((rv = ngram_fwdtree_reinit(ngs)) < 0)
277  return rv;
278  }
279  if (ngs->fwdflat) {
280  if ((rv = ngram_fwdflat_reinit(ngs)) < 0)
281  return rv;
282  }
283 
284  return rv;
285 }

346 static void
347 set_real_wid(ngram_search_t *ngs, int32 bp)
348 {
349  bptbl_t *ent, *prev;
350 
351  assert(bp != NO_BP);
352  ent = ngs->bp_table + bp;
353  if (ent->bp == NO_BP)
354  prev = NULL;
355  else
356  prev = ngs->bp_table + ent->bp;
357 
358  /* Propagate lm state for fillers, rotate it for words. */
359  if (dict_filler_word(ps_search_dict(ngs), ent->wid)) {
360  if (prev != NULL) {
361  ent->real_wid = prev->real_wid;
362  ent->prev_real_wid = prev->prev_real_wid;
363  }
364  else {
365  ent->real_wid = dict_basewid(ps_search_dict(ngs),
366  ent->wid);
367  ent->prev_real_wid = BAD_S3WID;
368  }
369  }
370  else {
371  ent->real_wid = dict_basewid(ps_search_dict(ngs), ent->wid);
372  if (prev != NULL)
373  ent->prev_real_wid = prev->real_wid;
374  else
375  ent->prev_real_wid = BAD_S3WID;
376  }
377 }

381 void
382 ngram_search_save_bp(ngram_search_t *ngs, int frame_idx,
383  int32 w, int32 score, int32 path, int32 rc)
384 {
385  int32 bp;
386 
387  /* Look for an existing exit for this word in this frame. The
388  * only reason one would exist is from a different right context
389  * triphone, but of course that happens quite frequently. */
390  bp = ngs->word_lat_idx[w];
391  if (bp != NO_BP) {
392 
393  if (frame_idx - ngs->bp_table[path].frame > NGRAM_HISTORY_LONG_WORD) {
394  E_WARN("Word '%s' survived for %d frames, potential overpruning\n", dict_wordstr(ps_search_dict(ngs), w),
395  frame_idx - ngs->bp_table[path].frame);
396  }
397 
398  /* Keep only the best scoring one, we will reconstruct the
399  * others from the right context scores - usually the history
400  * is not lost. */
401  if (ngs->bp_table[bp].score WORSE_THAN score) {
402  assert(path != bp); /* Pathological. */
403  if (ngs->bp_table[bp].bp != path) {
404  int32 bplh[2], newlh[2];
405  /* But, sometimes, the history *is* lost. If we wanted to
406  * do exact language model scoring we'd have to preserve
407  * these alternate histories. */
408  E_DEBUG(2,("Updating path history %d => %d frame %d\n",
409  ngs->bp_table[bp].bp, path, frame_idx));
410  bplh[0] = ngs->bp_table[bp].bp == -1
411  ? -1 : ngs->bp_table[ngs->bp_table[bp].bp].prev_real_wid;
412  bplh[1] = ngs->bp_table[bp].bp == -1
413  ? -1 : ngs->bp_table[ngs->bp_table[bp].bp].real_wid;
414  newlh[0] = path == -1
415  ? -1 : ngs->bp_table[path].prev_real_wid;
416  newlh[1] = path == -1
417  ? -1 : ngs->bp_table[path].real_wid;
418  /* Actually it's worth checking how often the actual
419  * language model state changes. */
420  if (bplh[0] != newlh[0] || bplh[1] != newlh[1]) {
421  /* It's fairly rare that the actual language model
422  * state changes, but it does happen some
423  * times. */
424  E_DEBUG(1, ("Updating language model state %s,%s => %s,%s frame %d\n",
425  dict_wordstr(ps_search_dict(ngs), bplh[0]),
426  dict_wordstr(ps_search_dict(ngs), bplh[1]),
427  dict_wordstr(ps_search_dict(ngs), newlh[0]),
428  dict_wordstr(ps_search_dict(ngs), newlh[1]),
429  frame_idx));
430  set_real_wid(ngs, bp);
431  }
432  ngs->bp_table[bp].bp = path;
433  }
434  ngs->bp_table[bp].score = score;
435  }
436  /* But do keep track of scores for all right contexts, since
437  * we need them to determine the starting path scores for any
438  * successors of this word exit. */
439  if (ngs->bp_table[bp].s_idx != -1)
440  ngs->bscore_stack[ngs->bp_table[bp].s_idx + rc] = score;
441  }
442  else {
443  int32 i, rcsize;
444  bptbl_t *be;
445 
446  /* This might happen if recognition fails. */
447  if (ngs->bpidx == NO_BP) {
448  E_ERROR("No entries in backpointer table!");
449  return;
450  }
451 
452  /* Expand the backpointer tables if necessary. */
453  if (ngs->bpidx >= ngs->bp_table_size) {
454  ngs->bp_table_size *= 2;
455  ngs->bp_table = ckd_realloc(ngs->bp_table,
456  ngs->bp_table_size
457  * sizeof(*ngs->bp_table));
458  E_INFO("Resized backpointer table to %d entries\n", ngs->bp_table_size);
459  }
460  if (ngs->bss_head >= ngs->bscore_stack_size
461  - bin_mdef_n_ciphone(ps_search_acmod(ngs)->mdef)) {
462  ngs->bscore_stack_size *= 2;
463  ngs->bscore_stack = ckd_realloc(ngs->bscore_stack,
464  ngs->bscore_stack_size
465  * sizeof(*ngs->bscore_stack));
466  E_INFO("Resized score stack to %d entries\n", ngs->bscore_stack_size);
467  }
468 
469  ngs->word_lat_idx[w] = ngs->bpidx;
470  be = &(ngs->bp_table[ngs->bpidx]);
471  be->wid = w;
472  be->frame = frame_idx;
473  be->bp = path;
474  be->score = score;
475  be->s_idx = ngs->bss_head;
476  be->valid = TRUE;
477  assert(path != ngs->bpidx);
478 
479  /* DICT2PID */
480  /* Get diphone ID for final phone and number of ssids corresponding to it. */
481  be->last_phone = dict_last_phone(ps_search_dict(ngs),w);
482  if (dict_is_single_phone(ps_search_dict(ngs), w)) {
483  be->last2_phone = -1;
484  be->s_idx = -1;
485  rcsize = 0;
486  }
487  else {
488  be->last2_phone = dict_second_last_phone(ps_search_dict(ngs),w);
489  rcsize = dict2pid_rssid(ps_search_dict2pid(ngs),
490  be->last_phone, be->last2_phone)->n_ssid;
491  }
492  /* Allocate some space on the bscore_stack for all of these triphones. */
493  for (i = 0; i < rcsize; ++i)
494  ngs->bscore_stack[ngs->bss_head + i] = WORST_SCORE;
495  if (rcsize)
496  ngs->bscore_stack[ngs->bss_head + rc] = score;
497  set_real_wid(ngs, ngs->bpidx);
498 
499  ngs->bpidx++;
500  ngs->bss_head += rcsize;
501  }

504 int
505 ngram_search_find_exit(ngram_search_t *ngs, int frame_idx, int32 *out_best_score, int32 *out_is_final)
506 {
507  /* End of backpointers for this frame. */
508  int end_bpidx;
509  int best_exit, bp;
510  int32 best_score;
511 
512  /* No hypothesis means no exit node! */
513  if (ngs->n_frame == 0)
514  return NO_BP;
515 
516  if (frame_idx == -1 || frame_idx >= ngs->n_frame)
517  frame_idx = ngs->n_frame - 1;
518  end_bpidx = ngs->bp_table_idx[frame_idx];
519 
520  best_score = WORST_SCORE;
521  best_exit = NO_BP;
522 
523  /* Scan back to find a frame with some backpointers in it. */
524  while (frame_idx >= 0 && ngs->bp_table_idx[frame_idx] == end_bpidx)
525  --frame_idx;
526  /* This is NOT an error, it just means there is no hypothesis yet. */
527  if (frame_idx < 0)
528  return NO_BP;
529 
530  /* Now find the entry for </s> OR the best scoring entry. */
531  assert(end_bpidx < ngs->bp_table_size);
532  for (bp = ngs->bp_table_idx[frame_idx]; bp < end_bpidx; ++bp) {

661 int32
662 ngram_search_exit_score(ngram_search_t *ngs, bptbl_t *pbe, int rcphone)
663 {

665  /* Get the mapping from right context phone ID to index in the

683 /*
684  * Compute acoustic and LM scores for a BPTable entry (segment).
685  */
686 void
687 ngram_compute_seg_score(ngram_search_t *ngs, bptbl_t *be, float32 lwf,
688  int32 *out_ascr, int32 *out_lscr)
689 {
690  bptbl_t *pbe;
691  int32 start_score;
692 
693  /* Start of utterance. */
694  if (be->bp == NO_BP) {
695  *out_ascr = be->score;
696  *out_lscr = 0;
697  return;
698  }
699 
700  /* Otherwise, calculate lscr and ascr. */
701  pbe = ngs->bp_table + be->bp;
702  start_score = ngram_search_exit_score(ngs, pbe,
703  dict_first_phone(ps_search_dict(ngs),be->wid));
704  assert(start_score BETTER_THAN WORST_SCORE);
705 
706  /* FIXME: These result in positive acoustic scores when filler
707  words have non-filler pronunciations. That whole business
708  is still pretty much broken but at least it doesn't
709  segfault. */
710  if (be->wid == ps_search_silence_wid(ngs)) {
711  *out_lscr = ngs->silpen;
712  }
713  else if (dict_filler_word(ps_search_dict(ngs), be->wid)) {

716  else {

719  be->real_wid,

721  pbe->prev_real_wid,
722  &n_used)>>SENSCR_SHIFT;
723  *out_lscr = *out_lscr * lwf;
724  }

1071 static void
1072 create_dag_nodes(ngram_search_t *ngs, ps_lattice_t *dag)
1073 {
1074  bptbl_t *bp_ptr;
1075  int32 i;
1076 
1077  for (i = 0, bp_ptr = ngs->bp_table; i < ngs->bpidx; ++i, ++bp_ptr) {
1078  int32 sf, ef, wid;
1079  ps_latnode_t *node;
1080 
1081  /* Skip invalid backpointers (these result from -maxwpf pruning) */
1082  if (!bp_ptr->valid)
1083  continue;
1084 
1085  sf = (bp_ptr->bp < 0) ? 0 : ngs->bp_table[bp_ptr->bp].frame + 1;
1086  ef = bp_ptr->frame;
1087  wid = bp_ptr->wid;
1088 
1089  assert(ef < dag->n_frames);
1090  /* Skip non-final </s> entries. */
1091  if ((wid == ps_search_finish_wid(ngs)) && (ef < dag->n_frames - 1))
1092  continue;
1093 
1094  /* Skip if word not in LM */
1095  if ((!dict_filler_word(ps_search_dict(ngs), wid))
1096  && (!ngram_model_set_known_wid(ngs->lmset,
1097  dict_basewid(ps_search_dict(ngs), wid))))
1098  continue;
1099 
1100  /* See if bptbl entry <wid,sf> already in lattice */
1101  for (node = dag->nodes; node; node = node->next) {
1102  if ((node->wid == wid) && (node->sf == sf))
1103  break;
1104  }
1105 
1106  /* For the moment, store bptbl indices in node.{fef,lef} */
1107  if (node)
1108  node->lef = i;
1109  else {
1110  /* New node; link to head of list */
1111  node = listelem_malloc(dag->latnode_alloc);
1112  node->wid = wid;
1113  node->sf = sf; /* This is a frame index. */
1114  node->fef = node->lef = i; /* These are backpointer indices (argh) */
1115  node->reachable = FALSE;
1116  node->entries = NULL;

1147 static ps_latnode_t *
1148 find_end_node(ngram_search_t *ngs, ps_lattice_t *dag, float32 lwf)
1149 {
1150  ps_latnode_t *node;
1151  int32 ef, bestbp, bp, bestscore;
1152 
1153  /* Find final node </s>.last_frame; nothing can follow this node */
1154  for (node = dag->nodes; node; node = node->next) {
1155  int32 lef = ngs->bp_table[node->lef].frame;
1156  if ((node->wid == ps_search_finish_wid(ngs))
1157  && (lef == dag->n_frames - 1))
1158  break;
1159  }
1160  if (node != NULL)
1161  return node;
1162

1164  * find the node corresponding to the best exit. */
1165  /* Find the last frame containing a word exit. */
1166  for (ef = dag->n_frames - 1;
1167  ef >= 0 && ngs->bp_table_idx[ef] == ngs->bpidx;
1168  --ef);

1170  E_ERROR("Empty backpointer table: can not build DAG.\n");
1171  return NULL;
1172  }
1173

1175  bestscore = WORST_SCORE;
1176  bestbp = NO_BP;
1177  for (bp = ngs->bp_table_idx[ef]; bp < ngs->bp_table_idx[ef + 1]; ++bp) {
1178  int32 n_used, l_scr, wid, prev_wid;
1179  wid = ngs->bp_table[bp].real_wid;
1180  prev_wid = ngs->bp_table[bp].prev_real_wid;
1181  /* Always prefer </s>, of which there will only be one per frame. */
1182  if (wid == ps_search_finish_wid(ngs)) {
1183  bestbp = bp;
1184  break;
1185  }

1187  wid, prev_wid, &n_used) >>SENSCR_SHIFT;
1188  l_scr = l_scr * lwf;
1189  if (ngs->bp_table[bp].score + l_scr BETTER_THAN bestscore) {
1190  bestscore = ngs->bp_table[bp].score + l_scr;
1191  bestbp = bp;
1192  }
1193  }
1194  if (bestbp == NO_BP) {
1195  E_ERROR("No word exits found in last frame (%d), assuming no recognition\n", ef);
1196  return NULL;
1197  }
1198  E_INFO("</s> not found in last frame, using %s.%d instead\n",
1199  dict_basestr(ps_search_dict(ngs), ngs->bp_table[bestbp].wid), ef);
1200

1202  for (node = dag->nodes; node; node = node->next) {
1203  if (node->lef == bestbp)
1204  return node;
1205  }
1206 
1207  /* FIXME: This seems to happen a lot! */
1208  E_ERROR("Failed to find DAG node corresponding to %s\n",
1209  dict_basestr(ps_search_dict(ngs), ngs->bp_table[bestbp].wid));
1210  return NULL;
1211 }

1214  * Build lattice from bptable.
1215  */
1216 ps_lattice_t *
1217 ngram_search_lattice(ps_search_t *search)
1218 {
1219  int32 i, score, ascr, lscr;
1220  ps_latnode_t *node, *from, *to;
1221  ngram_search_t *ngs;
1222  ps_lattice_t *dag;
1223  int min_endfr, nlink;
1224  float lwf;
1225 
1226  ngs = (ngram_search_t *)search;
1227  min_endfr = cmd_ln_int32_r(ps_search_config(search), "-min_endfr");
1228 
1229  /* If the best score is WORST_SCORE or worse, there is no way to
1230  * make a lattice. */
1231  if (ngs->best_score == WORST_SCORE || ngs->best_score WORSE_THAN WORST_SCORE)
1232  return NULL;
1233 
1234  /* Check to see if a lattice has previously been created over the
1235  * same number of frames, and reuse it if so. */
1236  if (search->dag && search->dag->n_frames == ngs->n_frame)
1237  return search->dag;
1238 
1239  /* Nope, create a new one. */
1240  ps_lattice_free(search->dag);
1241  search->dag = NULL;
1242  dag = ps_lattice_init_search(search, ngs->n_frame);
1243  /* Compute these such that they agree with the fwdtree language weight. */
1244  lwf = ngs->fwdflat ? ngs->fwdflat_fwdtree_lw_ratio : 1.0;
1245  create_dag_nodes(ngs, dag);
1246  if ((dag->start = find_start_node(ngs, dag)) == NULL)
1247  goto error_out;
1248  if ((dag->end = find_end_node(ngs, dag, ngs->bestpath_fwdtree_lw_ratio)) == NULL)
1249  goto error_out;
1250  E_INFO("lattice start node %s.%d end node %s.%d\n",
1251  dict_wordstr(search->dict, dag->start->wid), dag->start->sf,
1252  dict_wordstr(search->dict, dag->end->wid), dag->end->sf);
1253 
1254  ngram_compute_seg_score(ngs, ngs->bp_table + dag->end->lef, lwf,
1255  &dag->final_node_ascr, &lscr);
1256 
1257  /*
1258  * At this point, dag->nodes is ordered such that nodes earlier in
1259  * the list can follow (in time) those later in the list, but not
1260  * vice versa (see above - also note that adjacency is purely
1261  * determined by time which is why we can make this claim). Now
1262  * create precedence links and simultanesously mark all nodes that
1263  * can reach dag->end. (All nodes are reached from dag->start
1264  * simply by definition - they were created that way).
1265  *
1266  * Note that this also means that any nodes before dag->end in the
1267  * list can be discarded, meaning that dag->end will always be
1268  * equal to dag->nodes (FIXME: except when loading from a file but
1269  * we can fix that...)
1270  */
1271  i = 0;
1272  while (dag->nodes && dag->nodes != dag->end) {
1273  ps_latnode_t *next = dag->nodes->next;
1274  listelem_free(dag->latnode_alloc, dag->nodes);
1275  dag->nodes = next;
1276  ++i;
1277  }
1278  E_INFO("Eliminated %d nodes before end node\n", i);
1279  dag->end->reachable = TRUE;
1280  nlink = 0;
1281  for (to = dag->end; to; to = to->next) {
1282  int fef, lef;
1283 
1284  /* Skip if not reachable; it will never be reachable from dag->end */
1285  if (!to->reachable)
1286  continue;
1287 
1288  /* Prune nodes with too few endpoints - heuristic
1289  borrowed from Sphinx3 */
1290  fef = ngs->bp_table[to->fef].frame;
1291  lef = ngs->bp_table[to->lef].frame;
1292  if (to != dag->end && lef - fef < min_endfr) {
1293  to->reachable = FALSE;
1294  continue;
1295  }
1296 
1297  /* Find predecessors of to : from->fef+1 <= to->sf <= from->lef+1 */
1298  for (from = to->next; from; from = from->next) {
1299  bptbl_t *from_bpe;
1300 
1301  fef = ngs->bp_table[from->fef].frame;
1302  lef = ngs->bp_table[from->lef].frame;
1303 
1304  if ((to->sf <= fef) || (to->sf > lef + 1))
1305  continue;
1306  if (lef - fef < min_endfr) {
1307  assert(!from->reachable);
1308  continue;
1309  }
1310 
1311  /* Find bptable entry for "from" that exactly precedes "to" */
1312  i = from->fef;
1313  from_bpe = ngs->bp_table + i;
1314  for (; i <= from->lef; i++, from_bpe++) {
1315  if (from_bpe->wid != from->wid)
1316  continue;
1317  if (from_bpe->frame >= to->sf - 1)
1318  break;
1319  }
1320 
1321  if ((i > from->lef) || (from_bpe->frame != to->sf - 1))
1322  continue;
1323 
1324  /* Find acoustic score from.sf->to.sf-1 with right context = to */
1325  /* This gives us from_bpe's best acoustic score. */
1326  ngram_compute_seg_score(ngs, from_bpe, lwf,
1327  &ascr, &lscr);
1328  /* Now find the exact path score for from->to, including
1329  * the appropriate final triphone. In fact this might not
1330  * exist. */
1331  score = ngram_search_exit_score(ngs, from_bpe,
1332  dict_first_phone(ps_search_dict(ngs), to->wid));
1333  /* Does not exist. Can't create a link here. */
1334  if (score == WORST_SCORE)
1335  continue;
1336  /* Adjust the arc score to match the correct triphone. */
1337  else
1338  score = ascr + (score - from_bpe->score);
1339  if (score BETTER_THAN 0) {
1340  /* Scores must be negative, or Bad Things will happen.
1341  In general, they are, except in corner cases
1342  involving filler words. We don't want to throw any
1343  links away so we'll keep these, but with some
1344  arbitrarily improbable but recognizable score. */
1345  ps_lattice_link(dag, from, to, -424242, from_bpe->frame);
1346  ++nlink;
1347  from->reachable = TRUE;
1348  }
1349  else if (score BETTER_THAN WORST_SCORE) {
1350  ps_lattice_link(dag, from, to, score, from_bpe->frame);
1351  ++nlink;
1352  from->reachable = TRUE;
1353  }
1354  }
1355  }
1356 
1357  /* There must be at least one path between dag->start and dag->end */
1358  if (!dag->start->reachable) {
1359  E_ERROR("End node of lattice isolated; unreachable\n");
1360  goto error_out;
1361  }
1362 
1363  for (node = dag->nodes; node; node = node->next) {
1364  /* Change node->{fef,lef} from bptbl indices to frames. */
1365  node->fef = ngs->bp_table[node->fef].frame;
1366  node->lef = ngs->bp_table[node->lef].frame;
1367  /* Find base wid for nodes. */
1368  node->basewid = dict_basewid(search->dict, node->wid);
1369  }
1370 
1371  /* Link nodes with alternate pronunciations at the same timepoint. */
1372  for (node = dag->nodes; node; node = node->next) {
1373  ps_latnode_t *alt;
1374  /* Scan forward to find the next alternate, then stop. */
1375  for (alt = node->next; alt && alt->sf == node->sf; alt = alt->next) {
1376  if (alt->basewid == node->basewid) {
1377  alt->alt = node->alt;
1378  node->alt = alt;
1379  break;
1380  }
1381  }
1382  }
1383  E_INFO("Lattice has %d nodes, %d links\n", dag->n_nodes, nlink);
1384 
1385  /* Minor hack: If the final node is a filler word and not </s>,
1386  * then set its base word ID to </s>, so that the language model
1387  * scores won't be screwed up. */
1388  if (dict_filler_word(ps_search_dict(ngs), dag->end->wid))
1389  dag->end->basewid = ps_search_finish_wid(ngs);
1390 
1391  /* Free nodes unreachable from dag->end and their links */
1392  ps_lattice_delete_unreachable(dag);
1393 
1394  /* Build links around silence and filler words, since they do not
1395  * exist in the language model. */
1396  ps_lattice_bypass_fillers(dag, ngs->silpen, ngs->fillpen);
1397 
1398  search->dag = dag;
1399  return dag;
1400 
1401 error_out:
1402  ps_lattice_free(dag);
1403  return NULL;
1404 }
